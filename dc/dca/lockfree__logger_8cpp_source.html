172 auto lockfree_logger::start(void) -> std::optional<std::string>
173 {
174
175 // Set writer queues to the lock-free collector
176 lockfree_collector_->set_console_queue(console_writer_->get_job_queue());
177 lockfree_collector_->set_file_queue(file_writer_->get_job_queue());
178 lockfree_collector_->set_callback_queue(callback_writer_->get_job_queue());
179
180 // Start all components
181 if (auto error = lockfree_collector_->start(); error.has_error())
182 {
183 return formatter::format("Failed to start lock-free log collector: {}",
184 error.get_error().to_string());
185 }
186
187 if (auto error = console_writer_->start(); error.has_error())
188 {
189 lockfree_collector_->stop();
190 return formatter::format("Failed to start console writer: {}",
191 error.get_error().to_string());
192 }
193
194 if (auto error = file_writer_->start(); error.has_error())
195 {
196 console_writer_->stop();
197 lockfree_collector_->stop();
198 return formatter::format("Failed to start file writer: {}",
199 error.get_error().to_string());
200 }
201
202 if (auto error = callback_writer_->start(); error.has_error())
203 {
204 file_writer_->stop();
205 console_writer_->stop();
206 lockfree_collector_->stop();
207 return formatter::format("Failed to start callback writer: {}",
208 error.get_error().to_string());
209 }
210
211 return std::nullopt;
212 }
