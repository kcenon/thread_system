set(LIBRARY_NAME priority_thread_pool)

file(GLOB PRIORITY_THREAD_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)
file(GLOB PRIORITY_THREAD_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.tpp)

add_library(${LIBRARY_NAME} INTERFACE)
target_sources(${LIBRARY_NAME} INTERFACE ${PRIORITY_THREAD_HEADERS} ${PRIORITY_THREAD_SOURCES})
target_include_directories(${LIBRARY_NAME} INTERFACE ${CMAKE_CURRENT_SOURCE_DIR})
