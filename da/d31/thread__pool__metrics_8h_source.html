179 explicit job_tracker(thread_pool_metrics& metrics)
180 : metrics_(metrics)

182
183 metrics_.jobs_submitted_->increment();
184 metrics_.queue_depth_->increment();

190 void on_start() {
191 if (started_) return;
192 started_ = true;
193
194 start_time_ = std::chrono::steady_clock::now();
195 auto wait_duration = std::chrono::duration<double>(
196 start_time_ - enqueue_time_
197 ).count();
198
199 metrics_.queue_wait_time_->observe(wait_duration);
200 metrics_.queue_depth_->decrement();
201 metrics_.workers_active_->increment();
202 metrics_.workers_idle_->decrement();
203 }
204
208 void on_complete() {
209 if (completed_) return;
210 completed_ = true;
211
212 if (started_) {
213 auto duration = std::chrono::duration<double>(
214 std::chrono::steady_clock::now() - start_time_
215 ).count();
216
217 metrics_.job_duration_->observe(duration);
218 metrics_.jobs_completed_->increment();
219 metrics_.workers_active_->decrement();
220 metrics_.workers_idle_->increment();
221
222 // Update throughput
223 static thread_local auto last_throughput_update =
224 std::chrono::steady_clock::now();
225 static thread_local uint64_t jobs_in_window = 0;
226
227 jobs_in_window++;
228 auto now = std::chrono::steady_clock::now();
229 auto elapsed = std::chrono::duration<double>(
230 now - last_throughput_update
231 ).count();
232
233 if (elapsed >= 1.0) {
234 metrics_.throughput_->observe(jobs_in_window / elapsed);
235 jobs_in_window = 0;
236 last_throughput_update = now;
237 }
238 }
239 }
240
244 void on_error() {
245 if (completed_) return;
246 completed_ = true;
247
248 metrics_.jobs_failed_->increment();
249
250 if (started_) {
251 metrics_.workers_active_->decrement();
252 metrics_.workers_idle_->increment();
253 } else {
254 metrics_.queue_depth_->decrement();
255 }
256 }
257
261 void on_cancel() {
262 if (completed_) return;
263 completed_ = true;
264
265 metrics_.jobs_cancelled_->increment();
266
267 if (!started_) {
268 metrics_.queue_depth_->decrement();
269 }
270 }
271
272 ~job_tracker() {
273 if (!completed_) {
274 on_error(); // Treat incomplete tracking as error
275 }
276 }
