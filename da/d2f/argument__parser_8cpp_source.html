103  template <typename NumericType>
104  auto argument_manager::to_numeric(std::string_view key) const -> std::optional<NumericType>
105  {
106  auto val = to_string(key);
107  if (!val)
108  return std::nullopt;
109 
110  const std::string& str_val = val.value();
111  NumericType result{};
112  auto [ptr, ec] = std::from_chars(str_val.data(), str_val.data() + str_val.size(), result);
113 
114  if (ec == std::errc::invalid_argument || ec == std::errc::result_out_of_range)
115  return std::nullopt;
116  if (ptr != str_val.data() + str_val.size())
117  return std::nullopt; // Extra characters remain after conversion
118 
119  return result;
120  }

122  template <typename StringType>
123  auto argument_manager::try_parse(const StringType& arguments) -> std::optional<std::string>
124  {
125  if (arguments.empty())
126  {
127  return "no valid arguments found.";
128  }
129 
130  std::optional<std::string> converted;
131  std::optional<std::string> convert_error;
132 
133  // Handle string conversion
134  if constexpr (std::is_same_v<StringType, std::string>
135  || std::is_same_v<StringType, const char*>)
136  {
137  converted = std::string(arguments);
138  }
139  else if constexpr (std::is_same_v<StringType, std::wstring>
140  || std::is_same_v<StringType, const wchar_t*>)
141  {
142  std::tie(converted, convert_error) = convert_string::to_string(std::wstring(arguments));
143  if (!converted)
144  {
145  return convert_error ? convert_error
146  : std::string("Failed to convert wide string to string");
147  }
148  }
149  else
150  {
151  return "Unsupported string type";
152  }
153 
154  auto argument_string = converted.value();
155 
156  // Check if string is only whitespace
157  bool only_whitespace
158  = std::all_of(argument_string.begin(), argument_string.end(),
159  [](unsigned char c) { return std::isspace(c) || c == '\0'; });
160  if (only_whitespace)
161  {
162  return "no valid arguments found.";
163  }
164 
165  // Remove null terminators if any
166  auto null_pos = argument_string.find('\0');
167  if (null_pos != std::string::npos)
168  {
169  argument_string = argument_string.substr(0, null_pos);
170  }
171 
172  auto [splitted, split_error] = convert_string::split(argument_string, " ");
173  if (!splitted)
174  {
175  return split_error;
176  }
177 
178  auto splitted_vector = splitted.value();
179  // Remove empty or whitespace-only strings
180  splitted_vector.erase(
181  std::remove_if(
182  splitted_vector.begin(), splitted_vector.end(), [](const std::string& s)
183  { return s.empty() || s.find_first_not_of(" \t\n\r\0") == std::string::npos; }),
184  splitted_vector.end());
185 
186  if (splitted_vector.empty())
187  {
188  return "no valid arguments found.";
189  }
190 
191  auto [parsed, parse_error] = parse(splitted_vector);
192  if (parse_error)
193  {
194  return parse_error;
195  }
196 
197  arguments_ = parsed.value();
198  return std::nullopt;
199  }

201  template <typename CharType>
202  auto argument_manager::try_parse(int argc, CharType* argv[]) -> std::optional<std::string>
203  {
204  if (argc < 1)
205  {
206  return "Invalid argument count";
207  }
208 
209  std::vector<std::string> args;
210  bool found_valid = false;
211 
212  for (int i = 0; i < argc; ++i)
213  {
214  if (!argv[i])
215  {
216  return "Null argument pointer encountered";
217  }
218 
219  std::string arg;
220  if constexpr (std::is_same_v<CharType, wchar_t>)
221  {
222  auto [converted, error] = convert_string::to_string(std::wstring(argv[i]));
223  if (error)
224  {
225  return error;
226  }
227  arg = converted.value();
228  }
229  else
230  {
231  arg = std::string(argv[i]);
232  }
233 
234  if (!arg.empty() && arg.starts_with("--"))
235  {
236  found_valid = true;
237  }
238  args.push_back(std::move(arg));
239  }
240 
241  if (!found_valid)
242  {
243  return "no valid arguments found.";
244  }
245 
246  auto [parsed, error] = parse(args);
247  if (error)
248  {
249  return error;
250  }
251 
252  arguments_ = parsed.value();
253  return std::nullopt;
254  }

256  auto argument_manager::parse(const std::vector<std::string>& arguments)
257  -> std::tuple<std::optional<std::map<std::string, std::string>>, std::optional<std::string>>
258  {
259  if (arguments.empty())
260  {
261  return { std::nullopt, "no valid arguments found." };
262  }
263 
264  // Validate first argument or adjust start index
265  size_t start_index = 0;
266  if (!arguments[0].starts_with("--"))
267  {
268  if (arguments.size() == 1)
269  {
270  return { std::nullopt, "invalid argument: " + arguments[0] };
271  }
272  if (!arguments[1].starts_with("--"))
273  {
274  return { std::nullopt, "invalid argument: " + arguments[1] };
275  }
276  start_index = 1;
277  }
278 
279  std::map<std::string, std::string> result;
280  bool found_valid = false;
281 
282  for (size_t i = start_index; i < arguments.size(); ++i)
283  {
284  const auto& arg = arguments[i];
285 
286  if (!arg.starts_with("--"))
287  {
288  return { std::nullopt, "invalid argument: " + arg };
289  }
290 
291  if (arg == "--help")
292  {
293  // Help option: no value required
294  result[arg] = "display help";
295  found_valid = true;
296  continue;
297  }
298 
299  if (i + 1 >= arguments.size() || arguments[i + 1].starts_with("--"))
300  {
301  return { std::nullopt, "argument '" + arg + "' expects a value." };
302  }
303 
304  result[arg] = arguments[i + 1];
305  found_valid = true;
306  ++i;
307  }
308 
309  if (!found_valid)
310  {
311  return { std::nullopt, "no valid arguments found." };
312  }
313 
314  return { result, std::nullopt };
315  }
