58 auto lockfree_thread_pool::start(void) -> std::optional<std::string>
59 {
60 bool expected = false;
61 if (!start_pool_.compare_exchange_strong(expected, true))
62 {
63 return formatter::format("Lockfree thread pool '{}' is already started",
64 thread_title_);
65 }
66
67 // Start all workers
68 std::lock_guard<std::mutex> lock(workers_mutex_);
69 for (auto& worker : workers_)
70 {
71 if (!worker)
72 {
73 continue;
74 }
75
76 worker->set_job_queue(job_queue_);
77 auto result = worker->start();
78 if (result.has_error())
79 {
80 // If any worker fails to start, stop all previously started workers
81 stop(true);
82 return formatter::format("Failed to start worker in pool '{}': {}",
83 thread_title_, result.get_error().message());
84 }
85 }
86
87 return std::nullopt;
88 }

140 auto lockfree_thread_pool::enqueue(std::unique_ptr<lockfree_thread_worker>&& worker)
141 -> std::optional<std::string>
142 {
143 if (!worker)
144 {
145 return std::string("Cannot enqueue null worker");
146 }
147
148 std::lock_guard<std::mutex> lock(workers_mutex_);
149
150 // If pool is already started, start the new worker immediately
151 if (start_pool_.load())
152 {
153 worker->set_job_queue(job_queue_);
154 auto result = worker->start();
155 if (result.has_error())
156 {
157 return formatter::format("Failed to start worker: {}",
158 result.get_error().message());
159 }
160 }
161
162 workers_.push_back(std::move(worker));
163 return std::nullopt;

166 auto lockfree_thread_pool::enqueue_batch(
167 std::vector<std::unique_ptr<lockfree_thread_worker>>&& workers)
168 -> std::optional<std::string>
169 {
170 if (workers.empty())
171 {
172 return std::nullopt;
173 }
174
175 std::lock_guard<std::mutex> lock(workers_mutex_);
176
177 // If pool is already started, start all new workers
178 if (start_pool_.load())
179 {
180 for (auto& worker : workers)
181 {
182 if (!worker)
183 {
184 continue;
185 }
186
187 worker->set_job_queue(job_queue_);
188 auto result = worker->start();
189 if (result.has_error())
190 {
191 return formatter::format("Failed to start worker: {}",
192 result.get_error().message());
193 }
194 }
195 }
196
197 // Move all workers to the pool
198 for (auto& worker : workers)
199 {
200 if (worker)
201 {
202 workers_.push_back(std::move(worker));
203 }
204 }
205
206 return std::nullopt;

209 auto lockfree_thread_pool::stop(bool immediately) -> std::optional<std::string>
210 {
211 bool expected = true;
212 if (!start_pool_.compare_exchange_strong(expected, false))
213 {
214 return formatter::format("Lockfree thread pool '{}' is not started",
215 thread_title_);
216 }
217
218 // Clear the queue if immediate stop is requested
219 if (immediately && job_queue_)
220 {
221 job_queue_->clear();
222 }
223
224 // Stop all workers
225 std::lock_guard<std::mutex> lock(workers_mutex_);
226 std::vector<std::string> errors;
227
228 for (auto& worker : workers_)
229 {
230 if (!worker)
231 {
232 continue;
233 }
234
235 auto result = worker->stop();
236 if (result.has_error())
237 {
238 errors.push_back(result.get_error().message());
239 }
240 }
241
242 if (!errors.empty())
243 {
244 std::string error_msg = "Errors occurred while stopping workers: ";
245 for (size_t i = 0; i < errors.size(); ++i)
246 {
247 if (i > 0) error_msg += "; ";
248 error_msg += errors[i];
249 }
250 return error_msg;
251 }
252
253 return std::nullopt;
254 }
