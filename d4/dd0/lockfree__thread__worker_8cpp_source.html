117 auto lockfree_thread_worker::do_work(void) -> result_void
118 {
119 if (!job_queue_)
120 {
121 return result_void(error(error_code::invalid_argument,
122 "Job queue not set for lockfree worker"));
123 }
124
125 [[maybe_unused]] auto start_time = use_time_tag_ ? std::chrono::steady_clock::now()
126 : std::chrono::steady_clock::time_point{};
127
128 // Try batch processing if enabled
129 if (batch_processing_enabled_.load())
130 {
131 size_t processed = process_batch();
132 if (processed > 0)
133 {
134 current_backoff_ = backoff_config_.min_backoff;
135 return {};
136 }
137 }
138
139 // Try single job dequeue
140 auto dequeue_result = job_queue_->dequeue();
141
142 if (dequeue_result.has_value())
143 {
144 auto job = std::move(dequeue_result.value());
145
146 if (use_time_tag_)
147 {
148 auto process_start = std::chrono::steady_clock::now();
149 auto idle_duration = process_start - last_job_time_;
150 stats_.idle_time_ns.fetch_add(
151 static_cast<uint64_t>(std::chrono::duration_cast<std::chrono::nanoseconds>(idle_duration).count()));
152
153 auto result = process_job(std::move(job));
154
155 auto process_end = std::chrono::steady_clock::now();
156 auto process_duration = process_end - process_start;
157 stats_.total_processing_time_ns.fetch_add(
158 static_cast<uint64_t>(std::chrono::duration_cast<std::chrono::nanoseconds>(process_duration).count()));
159
160 last_job_time_ = process_end;
161 return result;
162 }
163 else
164 {
165 return process_job(std::move(job));
166 }
167 }
168 else if (!dequeue_result.has_value())
169 {
170 // Implement backoff strategy for contention
171 static size_t backoff_attempt = 0;
172 backoff(backoff_attempt++);
173
174 // Reset backoff counter after successful dequeue
175 if (backoff_attempt > backoff_config_.spin_count)
176 {
177 backoff_attempt = 0;
178 }
179 }
180
181 return {};
182 }

205 auto lockfree_thread_worker::after_stop(void) -> result_void
206 {
207 // Process any remaining jobs
208 while (job_queue_ && !job_queue_->empty())
209 {
210 auto dequeue_result = job_queue_->dequeue();
211 if (dequeue_result.has_value())
212 {
213 process_job(std::move(dequeue_result.value()));
214 }
215 }
216
217 return {};

220 auto lockfree_thread_worker::backoff(size_t attempt) -> void
221 {
222 if (attempt < backoff_config_.spin_count)
223 {
224 // Spin wait for initial attempts
225 size_t spin_count = 1ULL << attempt;
226 for (size_t i = 0; i < spin_count; ++i)
227 {
228 // Busy wait - prevent compiler optimization
229 std::atomic_thread_fence(std::memory_order_relaxed);
230 }
231 }
232 else
233 {
234 // Exponential backoff with sleep
235 stats_.backoff_count.fetch_add(1);
236
237 current_backoff_ = std::chrono::nanoseconds(
238 static_cast<int64_t>(current_backoff_.count() * backoff_config_.backoff_multiplier));
239
240 // Cap at maximum backoff
241 if (current_backoff_ > backoff_config_.max_backoff)
242 {
243 current_backoff_ = backoff_config_.max_backoff;
244 }
245
246 std::this_thread::sleep_for(current_backoff_);
247 }
248 }

250 auto lockfree_thread_worker::process_job(std::unique_ptr<job> job) -> result_void
251 {
252 if (!job)
253 {
254 return result_void(error(error_code::invalid_argument, "Null job in process_job"));
255 }
256
257 stats_.jobs_processed.fetch_add(1);
258
259 try
260 {
261 return job->do_work();
262 }
263 catch (const std::exception& e)
264 {
265 return result_void(error(error_code::job_execution_failed,
266 formatter::format("Job execution failed: {}", e.what())));
267 }
268 catch (...)
269 {
270 return result_void(error(error_code::job_execution_failed, "Unknown error in job execution"));
271 }
272 }

274 auto lockfree_thread_worker::process_batch() -> size_t
275 {
276 size_t processed = 0;
277 size_t batch_size = batch_size_.load();
278
279 std::vector<std::unique_ptr<job>> jobs;
280 jobs.reserve(batch_size);
281
282 // Try to dequeue multiple jobs
283 for (size_t i = 0; i < batch_size; ++i)
284 {
285 auto dequeue_result = job_queue_->dequeue();
286 if (dequeue_result.has_value())
287 {
288 jobs.push_back(std::move(dequeue_result.value()));
289 }
290 else
291 {
292 break;
293 }
294 }
295
296 if (!jobs.empty())
297 {
298 stats_.batch_dequeue_count.fetch_add(1);
299
300 // Process all dequeued jobs
301 for (auto& job : jobs)
302 {
303 process_job(std::move(job));
304 ++processed;
305 }
306 }
307
308 return processed;
309 }
