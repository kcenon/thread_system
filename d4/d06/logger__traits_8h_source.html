157 template<typename T>
158 std::string to_log_string(T&& value) {
159 if constexpr (is_string_like_v<std::decay_t<T>>) {
160 return std::string(std::forward<T>(value));
161 } else if constexpr (has_to_string_v<std::decay_t<T>>) {
162 return value.to_string();
163 } else if constexpr (std::is_arithmetic_v<std::decay_t<T>>) {
164 return std::to_string(value);
165 } else if constexpr (is_streamable_v<std::decay_t<T>>) {
166 std::ostringstream oss;
167 oss << value;
168 return oss.str();
169 } else {
170 static_assert(sizeof(T) == 0, "Type cannot be converted to string for logging");
171 }
172 }

196 public:
197 template<typename T>
198 message_eraser(T&& value)
199 requires Formattable<T>
200 : vtable_(&vtable_for<std::decay_t<T>>)
201 , storage_(std::forward<T>(value)) {}
202
203 std::string to_string() const {
204 return vtable_->to_string(storage_);
205 }
206
207 private:
208 struct vtable_t {
209 std::string (*to_string)(const void*);
210 void (*destroy)(void*);
211 };

213 template<typename T>
214 static constexpr vtable_t vtable_for = {
215 [](const void* ptr) { return detail::to_log_string(*static_cast<const T*>(ptr)); },
216 [](void* ptr) { static_cast<T*>(ptr)->~T(); }
217 };
218
219 const vtable_t* vtable_;
220 alignas(std::max_align_t) char storage_[64];
221 };

284 template<size_t BufferSize = 256>
285 class stack_string_builder {
286 char buffer_[BufferSize];
287 size_t pos_ = 0;
288
289 public:
290 template<typename T>
291 void append(T&& value) {
292 std::string str = to_log_string(std::forward<T>(value));
293 size_t copy_size = std::min(str.size(), BufferSize - pos_ - 1);
294 std::memcpy(buffer_ + pos_, str.data(), copy_size);
295 pos_ += copy_size;
296 buffer_[pos_] = '\0';
297 }
298
299 std::string to_string() const {
300 return std::string(buffer_, pos_);
301 }
302 };
