126 template<typename T>
127 std::string to_log_string(T&& value) {
128 if constexpr (is_string_like_v<std::decay_t<T>>) {
129 return std::string(std::forward<T>(value));
130 } else if constexpr (has_to_string_v<std::decay_t<T>>) {
131 return value.to_string();
132 } else if constexpr (std::is_arithmetic_v<std::decay_t<T>>) {
133 return std::to_string(value);
134 } else if constexpr (is_streamable_v<std::decay_t<T>>) {
135 std::ostringstream oss;
136 oss << value;
137 return oss.str();
138 } else {
139 static_assert(sizeof(T) == 0, "Type cannot be converted to string for logging");
140 }
141 }
