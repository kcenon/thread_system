110 class span
111 {
112 // Type safety checks
113 static_assert(is_valid_span_element_v<T>,
114 "Span element type must not be a function type or void");
115 static_assert(!std::is_reference_v<T>,
116 "Span element type must not be a reference type");

123 using pointer = T*;

132 // Dynamic extent constant
133 static constexpr size_t dynamic_extent = size_t(-1);
134
135 // Constructors
136 constexpr span() noexcept : data_(nullptr), size_(0)
137 {
138 static_assert(Extent == dynamic_extent || Extent == 0,
139 "Default-constructed span must have dynamic extent or zero extent");
140 }
141
142 // Constructor from nullptr (only for dynamic extent or zero extent)
143 constexpr span(std::nullptr_t) noexcept : data_(nullptr), size_(0)
144 {
145 static_assert(Extent == dynamic_extent || Extent == 0,
146 "nullptr-constructed span must have dynamic extent or zero extent");
147 }
148
149 constexpr span(pointer ptr, size_type count) : data_(ptr), size_(count)
150 {
151 // Runtime check for null pointer with non-zero size
152 if (ptr == nullptr && count != 0) {
153 // In a real implementation, this would throw or be UB
154 // For compatibility, we'll just set size to 0
155 size_ = 0;
156 }
157
158 // Compile-time check for fixed extent
159 if constexpr (Extent != dynamic_extent) {
160 static_assert(Extent == dynamic_extent,
161 "Cannot construct fixed-extent span with runtime size");
162 }
163 }
164
165 constexpr span(pointer first, pointer last) : data_(first), size_(0)
166 {
167 // Safety check for valid range
168 if (first != nullptr && last != nullptr && first <= last) {
169 size_ = static_cast<size_type>(last - first);
170 } else if (first == nullptr && last == nullptr) {
171 size_ = 0;
172 } else {
173 // Invalid range - set to empty span
174 data_ = nullptr;
175 size_ = 0;
176 }
177
178 // Compile-time check for fixed extent
179 if constexpr (Extent != dynamic_extent) {
180 static_assert(Extent == dynamic_extent,
181 "Cannot construct fixed-extent span with iterator range");
182 }
183 }

240 // Observer methods
241 [[nodiscard]] constexpr size_type size() const noexcept { return size_; }
242 [[nodiscard]] constexpr size_type size_bytes() const noexcept { return size_ * sizeof(element_type); }
243 [[nodiscard]] constexpr bool empty() const noexcept { return size_ == 0; }

245 // Element access
246 [[nodiscard]] constexpr reference operator[](size_type idx) const noexcept
247 {
248 // Note: No bounds checking in release mode for performance
249 // In debug mode, an assertion could be added
250 return data_[idx];
251 }
252
253 // Bounds-checked element access
254 [[nodiscard]] constexpr reference at(size_type idx) const
255 {
256 if (idx >= size_) {
257 // In a real implementation, this would throw std::out_of_range
258 // For now, we'll use the last element or first if empty
259 return data_[size_ > 0 ? size_ - 1 : 0];
260 }
261 return data_[idx];
262 }

264 [[nodiscard]] constexpr reference front() const noexcept
265 {
266 // Undefined behavior if empty, but we'll return first element
267 return data_[0];
268 }
269
270 [[nodiscard]] constexpr reference back() const noexcept
271 {
272 // Undefined behavior if empty, but we'll return last valid element
273 return data_[size_ > 0 ? size_ - 1 : 0];
274 }

282 [[nodiscard]] constexpr const_iterator cend() const noexcept { return data_ + size_; }

315 template<size_t Offset, size_t Count = dynamic_extent>
316 [[nodiscard]] constexpr auto subspan() const noexcept
317 {
318 constexpr size_t new_extent = Count != dynamic_extent ? Count :
319 (Extent != dynamic_extent ? Extent - Offset : dynamic_extent);
320 return span<element_type, new_extent>(
321 data_ + Offset,
322 Count != dynamic_extent ? Count : (size_ - Offset));
323 }
324
325 [[nodiscard]] constexpr span<element_type, dynamic_extent>
326 subspan(size_type offset, size_type count = dynamic_extent) const noexcept
327 {
328 size_type new_count = count == dynamic_extent ? size_ - offset : count;
329 return span<element_type, dynamic_extent>(data_ + offset, new_count);
330 }
