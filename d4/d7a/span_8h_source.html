265 template<size_t Offset, size_t Count = dynamic_extent>
266 [[nodiscard]] constexpr auto subspan() const noexcept
267 {
268 constexpr size_t new_extent = Count != dynamic_extent ? Count :
269 (Extent != dynamic_extent ? Extent - Offset : dynamic_extent);
270 return span<element_type, new_extent>(
271 data_ + Offset,
272 Count != dynamic_extent ? Count : (size_ - Offset));
273 }

277 {
278 size_type new_count = count == dynamic_extent ? size_ - offset : count;
279 return span<element_type, dynamic_extent>(data_ + offset, new_count);
280 }
