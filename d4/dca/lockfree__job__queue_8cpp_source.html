62 auto lockfree_job_queue::enqueue(std::unique_ptr<job>&& value) -> result_void
63 {
64 if (!value) {
65 return error{error_code::invalid_argument, "Cannot enqueue null job"};
66 }
67
68 if (stop_.load(std::memory_order_acquire)) {
69 return error{error_code::queue_stopped, "Queue is stopped"};
70 }
71
72 auto start_time = std::chrono::high_resolution_clock::now();
73
74 // Allocate storage for the job
75 job_ptr* data_storage = new job_ptr(std::move(value));
76
77 try {
78 auto result = enqueue_impl(data_storage);
79
80 auto duration = std::chrono::high_resolution_clock::now() - start_time;
81 record_enqueue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
82
83 return result;
84 } catch (...) {
85 delete data_storage;
86 return error{error_code::unknown_error, "Exception during enqueue"};
87 }

90 auto lockfree_job_queue::enqueue_batch(std::vector<std::unique_ptr<job>>&& jobs) -> result_void
91 {
92 if (jobs.empty()) {
93 return error{error_code::invalid_argument, "Cannot enqueue empty batch"};
94 }
95
96 if (stop_.load(std::memory_order_acquire)) {
97 return error{error_code::queue_stopped, "Queue is stopped"};
98 }
99
100 if (jobs.size() > MAX_BATCH_SIZE) {
101 return error{error_code::invalid_argument, "Batch size exceeds maximum"};
102 }
103
104 auto start_time = std::chrono::high_resolution_clock::now();
105
106 // Prepare nodes and data storage
107 std::vector<Node*> nodes;
108 std::vector<job_ptr*> data_storage;
109
110 try {
111 nodes.reserve(jobs.size());
112 data_storage.reserve(jobs.size());
113
114 // Create linked list of nodes
115 for (size_t i = 0; i < jobs.size(); ++i) {
116 if (!jobs[i]) {
117 throw std::invalid_argument("Null job in batch");
118 }
119
120 Node* node = allocate_node();
121 if (!node) {
122 throw std::runtime_error("Failed to allocate node");
123 }
124 job_ptr* data = new job_ptr(std::move(jobs[i]));
125
126 node->set_data(data);
127 nodes.push_back(node);
128 data_storage.push_back(data);
129
130 if (i > 0 && nodes.size() > 1) {
131 nodes[i-1]->next.store(node, std::memory_order_release);
132 }
133 }
134
135 // Link the batch into the queue
136 Node* first_node = nodes.front();
137 Node* last_node = nodes.back();
138 size_t retry_count = 0;
139
140 while (true) {
141 Node* tail = tail_.load(std::memory_order_acquire);
142 Node* next = tail->next.load(std::memory_order_acquire);
143
144 if (tail == tail_.load(std::memory_order_acquire)) {
145 if (next == nullptr) {
146 // Try to link the batch
147 if (tail->next.compare_exchange_weak(next, first_node,
148 std::memory_order_release,
149 std::memory_order_relaxed)) {
150 // Try to advance tail
151 tail_.compare_exchange_weak(tail, last_node,
152 std::memory_order_release,
153 std::memory_order_relaxed);
154 break;
155 }
156 } else {
157 // Help advance tail
158 tail_.compare_exchange_weak(tail, next,
159 std::memory_order_release,
160 std::memory_order_relaxed);
161 }
162 }
163
164 if (++retry_count > RETRY_THRESHOLD) {
165 increment_retry_count();
166 retry_count = 0;
167 }
168 }
169
170 // Update statistics
171 stats_.enqueue_batch_count.fetch_add(1, std::memory_order_relaxed);
172 stats_.enqueue_count.fetch_add(jobs.size(), std::memory_order_relaxed);
173 stats_.current_size.fetch_add(jobs.size(), std::memory_order_relaxed);
174
175 auto duration = std::chrono::high_resolution_clock::now() - start_time;
176 record_enqueue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
177
178 return result_void{};
179
180 } catch (...) {
181 // Clean up on failure
182 for (auto* data : data_storage) {
183 delete data;
184 }
185 for (auto* node : nodes) {
186 deallocate_node(node);
187 }
188 return error{error_code::unknown_error, "Exception during batch enqueue"};
189 }
190 }

192 auto lockfree_job_queue::dequeue() -> result<std::unique_ptr<job>>
193 {
194 if (stop_.load(std::memory_order_acquire)) {
195 return error{error_code::queue_stopped, "Queue is stopped"};
196 }
197
198 auto start_time = std::chrono::high_resolution_clock::now();
199
200 auto result = dequeue_impl();
201
202 auto duration = std::chrono::high_resolution_clock::now() - start_time;
203 record_dequeue_time(std::chrono::duration_cast<std::chrono::nanoseconds>(duration));
204
205 if (result.has_value()) {
206 condition_.notify_one();
207 }
208
209 return result;
210 }

212 auto lockfree_job_queue::dequeue_batch() -> std::deque<std::unique_ptr<job>>
213 {
214 std::deque<std::unique_ptr<job>> result;
215
216 // Try to dequeue up to MAX_BATCH_SIZE items
217 for (size_t i = 0; i < MAX_BATCH_SIZE; ++i) {
218 auto job_result = try_dequeue();
219 if (!job_result.has_value()) {
220 break;
221 }
222 result.push_back(std::move(job_result.value()));
223 }
224
225 if (!result.empty()) {
226 stats_.dequeue_batch_count.fetch_add(1, std::memory_order_relaxed);
227 condition_.notify_all();
228 }
229
230 return result;
231 }

331 auto lockfree_job_queue::enqueue_impl(job_ptr* data_storage) -> result_void
332 {
333 Node* new_node = allocate_node();
334 if (!new_node) {
335 return error{error_code::resource_allocation_failed, "Failed to allocate node"};
336 }
337
338 size_t retry_count = 0;
339 size_t total_retries = 0;
340
341 while (total_retries < MAX_TOTAL_RETRIES) {
342 Node* tail = tail_.load(std::memory_order_acquire);
343 Node* next = tail->next.load(std::memory_order_acquire);
344
345 // Check if tail is still valid
346 if (tail == tail_.load(std::memory_order_acquire)) {
347 if (next == nullptr) {
348 // Try to link new node
349 if (tail->next.compare_exchange_weak(next, new_node,
350 std::memory_order_release,
351 std::memory_order_relaxed)) {
352 // Set data after linking
353 new_node->set_data(data_storage);
354
355 // Try to advance tail
356 tail_.compare_exchange_weak(tail, new_node,
357 std::memory_order_release,
358 std::memory_order_relaxed);
359
360 stats_.enqueue_count.fetch_add(1, std::memory_order_relaxed);
361 stats_.current_size.fetch_add(1, std::memory_order_relaxed);
362
363 return result_void{};
364 }
365 } else {
366 // Help advance tail
367 tail_.compare_exchange_weak(tail, next,
368 std::memory_order_release,
369 std::memory_order_relaxed);
370 }
371 }
372
373 if (++retry_count > RETRY_THRESHOLD) {
374 increment_retry_count();
375 retry_count = 0;
376 }
377
378 ++total_retries;
379
380 // Add small delay to reduce contention
381 if (total_retries % 100 == 0) {
382 std::this_thread::yield();
383 }
384 }
385
386 // If we reach here, we've exceeded max retries
387 deallocate_node(new_node);
388 return error{error_code::resource_limit_reached, "Enqueue failed after maximum retries"};
389 }

391 auto lockfree_job_queue::dequeue_impl() -> result<job_ptr>
392 {
393 size_t retry_count = 0;
394 size_t total_retries = 0;
395
396 while (total_retries < MAX_TOTAL_RETRIES) {
397 // Use hazard pointers to protect head access
398 auto hp = hp_manager_->acquire();
399 Node* head = hp.protect(head_);
400
401 if (!head) {
402 // This should never happen if constructor succeeded
403 return error{error_code::unknown_error, "Head pointer is null"};
404 }
405
406 Node* tail = tail_.load(std::memory_order_acquire);
407 Node* next = head->next.load(std::memory_order_acquire);
408
409 // Check if head is still valid
410 if (head == head_.load(std::memory_order_acquire)) {
411 if (head == tail) {
412 if (next == nullptr) {
413 // Queue is empty
414 return error{error_code::queue_empty, "Queue is empty"};
415 }
416
417 // Help advance tail
418 tail_.compare_exchange_weak(tail, next,
419 std::memory_order_release,
420 std::memory_order_relaxed);
421 } else {
422 // Read data from next node
423 if (next == nullptr) {
424 continue; // Retry
425 }
426
427 job_ptr* data = next->get_data(std::memory_order_acquire);
428
429 // Try to advance head
430 if (head_.compare_exchange_weak(head, next,
431 std::memory_order_release,
432 std::memory_order_relaxed)) {
433 if (data) {
434 auto result = std::move(*data);
435 delete data;
436
437 // Retire the old head node
438 retire_node(head);
439
440 stats_.dequeue_count.fetch_add(1, std::memory_order_relaxed);
441 stats_.current_size.fetch_sub(1, std::memory_order_relaxed);
442
443 return result;
444 } else {
445 // Data was null, retry
446 retire_node(head);
447 continue;
448 }
449 }
450 }
451 }
452
453 if (++retry_count > RETRY_THRESHOLD) {
454 increment_retry_count();
455 retry_count = 0;
456 }
457
458 ++total_retries;
459
460 // Add small delay to reduce contention
461 if (total_retries % 100 == 0) {
462 std::this_thread::yield();
463 }
464 }
465
466 // If we reach here, we've exceeded max retries
467 return error{error_code::resource_limit_reached, "Dequeue failed after maximum retries"};
468 }
