288 class scoped_timer {
289 public:
294 explicit scoped_timer(std::atomic<std::uint64_t>& target)
295 : target_(target), start_time_(std::chrono::steady_clock::now()) {}
296
303 ~scoped_timer() {
304 auto end_time = std::chrono::steady_clock::now();
305 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
306 end_time - start_time_).count();
307 target_.fetch_add(static_cast<std::uint64_t>(duration), std::memory_order_relaxed);
308 }
309
310 // Copy and move disabled
311 scoped_timer(const scoped_timer&) = delete;
312 scoped_timer& operator=(const scoped_timer&) = delete;
313 scoped_timer(scoped_timer&&) = delete;
314 scoped_timer& operator=(scoped_timer&&) = delete;
315
316 private:
317 std::atomic<std::uint64_t>& target_;
318 time_point start_time_;
319 };
