546 // Wait synchronously for the task to complete (blocking)
547 result_t<T> wait() const {
548 if (!valid()) {
549 return error{error_code::invalid_argument, "Cannot wait on an invalid task"};
550 }
551
552 // If the task is already completed, get the result immediately
553 if (handle_.promise().is_ready()) {
554 try {
555 if constexpr (std::is_void_v<T>) {
556 handle_.promise().get_result();
557 return result_void{};
558 } else {
559 return handle_.promise().get_result();
560 }
561 } catch (const std::exception& e) {
562 return error{error_code::job_execution_failed, e.what()};
563 }

566 // Structure for safe continuation & cleanup (especially important for Windows)
567 struct continuation_context {
568 std::promise<void> barrier;
569
570 static void invoke_and_destroy(void* ptr) {
571 auto* ctx = static_cast<continuation_context*>(ptr);
572 try {
573 ctx->barrier.set_value();
574 } catch (...) {
575 // Ignore any exceptions if the promise is already satisfied
576 }
577 delete ctx;
578 }
579 };
580
581 // Create a new continuation context with our barrier
582 auto* ctx = new continuation_context;
583 auto future = ctx->barrier.get_future();
584
585 // Set up the continuation
586 handle_.promise().set_continuation(
587 std::coroutine_handle<>::from_address(ctx)
588 );
589
590 // Wait for completion with a timeout
591 auto status = future.wait_for(std::chrono::seconds(5));
592
593 if (status == std::future_status::timeout) {
594 // Clean up our context on timeout to avoid memory leaks
595 delete ctx;
596 return error{error_code::operation_timeout, "Task wait timed out after 5 seconds"};
597 }
598
599 try {
600 if constexpr (std::is_void_v<T>) {
601 handle_.promise().get_result();
602 return result_void{};
603 } else {
604 return handle_.promise().get_result();
605 }
606 } catch (const std::exception& e) {
607 return error{error_code::job_execution_failed, e.what()};
608 }
609 }

647// Helper function to create a task from a callable
648template <typename Func>
649auto make_task(Func&& func) -> task<std::invoke_result_t<Func>> {
650 using result_type = std::invoke_result_t<Func>;
651
652#if defined(_MSC_VER)
653 // MSVC-specific implementation
654 // MSVC requires different approach for handling errors from coroutines
655 // We need to handle error conditions by wrapping the entire function
656 try {
657 if constexpr (std::is_void_v<result_type>) {
658 std::forward<Func>(func)();
659 co_return;
660 } else {
661 co_return std::forward<Func>(func)();
662 }
663 } catch (const error& e) {
664 // Special case for our error type - we already have the error information
665 if constexpr (std::is_void_v<result_type>) {
666 co_return;
667 } else if constexpr (detail::is_result_v<result_type>) {
668 co_return result_type(e);
669 } else {
670 co_return static_cast<result_type>(0);
671 }
672 } catch (const std::exception& e) {
673 // For standard exceptions, convert to our error type
674 if constexpr (std::is_void_v<result_type>) {
675 co_return;
676 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
677 std::is_same_v<result_type, result_void>) {
678 co_return result_type(error{error_code::job_execution_failed, e.what()});
679 } else if constexpr (detail::is_result_v<result_type>) {
680 co_return result_type(error{error_code::job_execution_failed, e.what()});
681 } else {
682 co_return static_cast<result_type>(0);
683 }
684 }
685#elif defined(THREAD_SYSTEM_MINGW) || defined(THREAD_SYSTEM_MSYS)
686 // MinGW/MSYS-specific implementation
687 // These environments need special handling similar to MSVC
688 try {
689 if constexpr (std::is_void_v<result_type>) {
690 std::forward<Func>(func)();
691 co_return;
692 } else {
693 co_return std::forward<Func>(func)();
694 }
695 } catch (const error& e) {
696 // Special case for our error type
697 if constexpr (std::is_void_v<result_type>) {
698 co_return;
699 } else if constexpr (detail::is_result_v<result_type>) {
700 co_return result_type(e);
701 } else {
702 co_return static_cast<result_type>(0);
703 }
704 } catch (const std::exception& e) {
705 // For standard exceptions
706 if constexpr (std::is_void_v<result_type>) {
707 co_return;
708 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
709 std::is_same_v<result_type, result_void>) {
710 co_return result_type(error{error_code::job_execution_failed, e.what()});
711 } else if constexpr (detail::is_result_v<result_type>) {
712 co_return result_type(error{error_code::job_execution_failed, e.what()});
713 } else {
714 co_return static_cast<result_type>(0);
715 }
716 }
717#else
718 // Implementation for Clang, GCC, etc.
719 try {
720 if constexpr (std::is_void_v<result_type>) {
721 std::forward<Func>(func)();
722 co_return;
723 } else {
724 co_return std::forward<Func>(func)();
725 }
726 } catch (const error& e) {
727 // Direct handling of our error type
728 #if defined(__clang__) && defined(__has_builtin) && __has_builtin(__builtin_coro_frame)
729 // Clang-specific implementation using __builtin_coro_frame if available
730 auto handle = std::coroutine_handle<task_promise<result_type>>::from_promise(
731 *reinterpret_cast<task_promise<result_type>*>(
732 std::coroutine_handle<>::from_address(
733 __builtin_coro_frame()
734 ).address()
735 )
736 );
737 handle.promise().set_error(e);
738 #else
739 // For other compilers, rethrow and handle in the catch block below
740 throw e;
741 #endif
742
743 // Fallback return values
744 if constexpr (std::is_void_v<result_type>) {
745 co_return;
746 } else if constexpr (detail::is_result_v<result_type>) {
747 co_return result_type(e);
748 } else {
749 co_return static_cast<result_type>(0);
750 }
751 } catch (const std::exception& e) {
752 // Can't use co_await in catch block, so manually handle the error
753 // We need a safe, cross-compiler way to access the promise
754 #if defined(__clang__) && defined(__has_builtin) && __has_builtin(__builtin_coro_frame)
755 // Clang-specific implementation using __builtin_coro_frame
756 auto handle = std::coroutine_handle<task_promise<result_type>>::from_promise(
757 *reinterpret_cast<task_promise<result_type>*>(
758 std::coroutine_handle<>::from_address(
759 __builtin_coro_frame()
760 ).address()
761 )
762 );
763 handle.promise().set_error(error{error_code::job_execution_failed, e.what()});
764 #elif defined(__GNUC__) || defined(__GNUG__)
765 // GCC implementation
766 // Convert to our error type then throw - the promise will catch it
767 throw error{error_code::job_execution_failed, e.what()};
768 #else
769 // Generic fallback for other compilers
770 throw error{error_code::job_execution_failed, e.what()};
771 #endif
772
773 // We need to provide a return value for non-void tasks
774 if constexpr (std::is_void_v<result_type>) {
775 co_return;
776 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
777 std::is_same_v<result_type, result_void>) {
778 // For result-based return types
779 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
780 } else if constexpr (detail::is_result_v<result_type>) {
781 // For any other result<T> types
782 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
783 } else {
784 // For regular return types - return default value of T
785 co_return static_cast<result_type>(0);
786 }
787 }
788#endif
789}

791// Simple helper function to sleep and resume the coroutine
792inline void sleep_and_resume(std::chrono::milliseconds duration, std::coroutine_handle<> handle) {
793 // Create a shared state to ensure we properly manage thread lifetime
794 struct shared_state {
795 std::thread timer_thread;
796 std::atomic<bool> completed{false};
797 std::atomic<bool> destroyed{false};
798
799 ~shared_state() {
800 destroyed.store(true);
801 if (timer_thread.joinable()) {
802 timer_thread.join();
803 }
804 }
805 };
806
807 // Using shared_ptr for automatic cleanup when all references are gone
808 auto state = std::make_shared<shared_state>();
809
810 // Create the timer thread that will sleep and then resume the coroutine
811 state->timer_thread = std::thread([state, duration, handle]() {
812 std::this_thread::sleep_for(duration);
813
814 // Only resume if we haven't been destroyed yet
815 if (!state->destroyed.load()) {
816 handle.resume();
817 }
818
819 state->completed.store(true);
820 });
821
822#if defined(THREAD_SYSTEM_WINDOWS)
823 // On Windows (including MSVC, MinGW, MSYS), we use a dedicated cleanup approach
824 std::thread cleanup_thread([state]() {
825 // This thread's only job is to hold a reference to the state
826 // until the timer thread completes to ensure proper cleanup
827 while (!state->completed.load()) {
828 std::this_thread::sleep_for(std::chrono::milliseconds(10));
829 }
830 });
831 cleanup_thread.detach();
832#else
833 // On non-Windows platforms, we can safely detach the thread
834 // The shared_ptr will keep the state alive until all references are gone
835 state->timer_thread.detach();
836
837 // Create a second detached thread as a safety net that will
838 // ensure the state lives long enough for the timer thread to finish
839 std::thread([state]() {
840 while (!state->completed.load()) {
841 std::this_thread::sleep_for(std::chrono::milliseconds(50));
842 }
843 }).detach();
844#endif
845}
