466 // Wait synchronously for the task to complete (blocking)
467 result_t<T> wait() const {
468 if (!valid()) {
469 return error{error_code::invalid_argument, "Cannot wait on an invalid task"};
470 }
471
472 // If the task is already completed, get the result immediately
473 if (handle_.promise().is_ready()) {
474 try {
475 if constexpr (std::is_void_v<T>) {
476 handle_.promise().get_result();
477 return result_void{};
478 } else {
479 return handle_.promise().get_result();
480 }
481 } catch (const std::exception& e) {
482 return error{error_code::job_execution_failed, e.what()};
483 }
484 }
485
486 // Create a latch to wait for completion
487 auto barrier_ptr = std::make_shared<std::promise<void>>();
488 auto future = barrier_ptr->get_future();
489
490 // Setup continuation to signal the latch using a heap allocated shared_ptr
491 auto continuation_fn = new auto([barrier_ptr]() {
492 try {
493 barrier_ptr->set_value();
494 } catch (...) {
495 // Ignore any exceptions if the promise is already satisfied
496 }
497 });
498
499 handle_.promise().set_continuation(
500 std::coroutine_handle<>::from_address(reinterpret_cast<void*>(continuation_fn))
501 );
502
503 // Wait for completion with a timeout
504 auto status = future.wait_for(std::chrono::seconds(5));
505
506 if (status == std::future_status::timeout) {
507 return error{error_code::operation_timeout, "Task wait timed out after 5 seconds"};
508 }
509
510 try {
511 if constexpr (std::is_void_v<T>) {
512 handle_.promise().get_result();
513 return result_void{};
514 } else {
515 return handle_.promise().get_result();
516 }
517 } catch (const std::exception& e) {
518 return error{error_code::job_execution_failed, e.what()};
519 }

558// Helper function to create a task from a callable
559template <typename Func>
560auto make_task(Func&& func) -> task<std::invoke_result_t<Func>> {
561 using result_type = std::invoke_result_t<Func>;
562
563 try {
564 if constexpr (std::is_void_v<result_type>) {
565 std::forward<Func>(func)();
566 co_return;
567 } else {
568 co_return std::forward<Func>(func)();
569 }
570 } catch (const std::exception& e) {
571 // Can't use co_await in catch block, so create error directly
572 auto handle = std::coroutine_handle<task_promise<result_type>>::from_promise(
573 *reinterpret_cast<task_promise<result_type>*>(
574 std::coroutine_handle<>::from_address(
575 __builtin_coro_frame()
576 ).address()
577 )
578 );
579 handle.promise().set_error(error{error_code::job_execution_failed, e.what()});
580
581 // We need to provide a return value for non-void tasks
582 if constexpr (std::is_void_v<result_type>) {
583 co_return;
584 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
585 std::is_same_v<result_type, result_void>) {
586 // For result-based return types
587 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
588 } else if constexpr (detail::is_result_v<result_type>) {
589 // For any other result<T> types
590 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
591 } else {
592 // For regular return types - return default value of T
593 co_return static_cast<result_type>(0);
594 }
595 }
