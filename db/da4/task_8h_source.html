617template <typename Func>
618auto make_task(Func&& func) -> task<std::invoke_result_t<Func>> {
619 using result_type = std::invoke_result_t<Func>;
620
621#if defined(_MSC_VER)
622 // MSVC-specific implementation
623 // MSVC requires different approach for handling errors from coroutines
624 // We need to handle error conditions by wrapping the entire function
625 try {
626 if constexpr (std::is_void_v<result_type>) {
627 std::forward<Func>(func)();
628 co_return;
629 } else {
630 co_return std::forward<Func>(func)();
631 }
632 } catch (const error& e) {
633 // Special case for our error type - we already have the error information
634 if constexpr (std::is_void_v<result_type>) {
635 co_return;
636 } else if constexpr (detail::is_result_v<result_type>) {
637 co_return result_type(e);
638 } else {
639 co_return static_cast<result_type>(0);
640 }
641 } catch (const std::exception& e) {
642 // For standard exceptions, convert to our error type
643 if constexpr (std::is_void_v<result_type>) {
644 co_return;
645 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
646 std::is_same_v<result_type, result_void>) {
647 co_return result_type(error{error_code::job_execution_failed, e.what()});
648 } else if constexpr (detail::is_result_v<result_type>) {
649 co_return result_type(error{error_code::job_execution_failed, e.what()});
650 } else {
651 co_return static_cast<result_type>(0);
652 }
653 }
654#elif defined(THREAD_SYSTEM_MINGW) || defined(THREAD_SYSTEM_MSYS)
655 // MinGW/MSYS-specific implementation
656 // These environments need special handling similar to MSVC
657 try {
658 if constexpr (std::is_void_v<result_type>) {
659 std::forward<Func>(func)();
660 co_return;
661 } else {
662 co_return std::forward<Func>(func)();
663 }
664 } catch (const error& e) {
665 // Special case for our error type
666 if constexpr (std::is_void_v<result_type>) {
667 co_return;
668 } else if constexpr (detail::is_result_v<result_type>) {
669 co_return result_type(e);
670 } else {
671 co_return static_cast<result_type>(0);
672 }
673 } catch (const std::exception& e) {
674 // For standard exceptions
675 if constexpr (std::is_void_v<result_type>) {
676 co_return;
677 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
678 std::is_same_v<result_type, result_void>) {
679 co_return result_type(error{error_code::job_execution_failed, e.what()});
680 } else if constexpr (detail::is_result_v<result_type>) {
681 co_return result_type(error{error_code::job_execution_failed, e.what()});
682 } else {
683 co_return static_cast<result_type>(0);
684 }
685 }
686#else
687 // Implementation for Clang, GCC, etc.
688 try {
689 if constexpr (std::is_void_v<result_type>) {
690 std::forward<Func>(func)();
691 co_return;
692 } else {
693 co_return std::forward<Func>(func)();
694 }
695 } catch (const error& e) {
696 // Direct handling of our error type
697 #if defined(__clang__) && defined(__has_builtin) && __has_builtin(__builtin_coro_frame)
698 // Clang-specific implementation using __builtin_coro_frame if available
699 auto handle = std::coroutine_handle<task_promise<result_type>>::from_promise(
700 *reinterpret_cast<task_promise<result_type>*>(
701 std::coroutine_handle<>::from_address(
702 __builtin_coro_frame()
703 ).address()
704 )
705 );
706 handle.promise().set_error(e);
707 #else
708 // For other compilers, rethrow and handle in the catch block below
709 throw e;
710 #endif
711
712 // Fallback return values
713 if constexpr (std::is_void_v<result_type>) {
714 co_return;
715 } else if constexpr (detail::is_result_v<result_type>) {
716 co_return result_type(e);
717 } else {
718 co_return static_cast<result_type>(0);
719 }
720 } catch (const std::exception& e) {
721 // Can't use co_await in catch block, so manually handle the error
722 // We need a safe, cross-compiler way to access the promise
723 #if defined(__clang__) && defined(__has_builtin) && __has_builtin(__builtin_coro_frame)
724 // Clang-specific implementation using __builtin_coro_frame
725 auto handle = std::coroutine_handle<task_promise<result_type>>::from_promise(
726 *reinterpret_cast<task_promise<result_type>*>(
727 std::coroutine_handle<>::from_address(
728 __builtin_coro_frame()
729 ).address()
730 )
731 );
732 handle.promise().set_error(error{error_code::job_execution_failed, e.what()});
733 #elif defined(__GNUC__) || defined(__GNUG__)
734 // GCC implementation
735 // Convert to our error type then throw - the promise will catch it
736 throw error{error_code::job_execution_failed, e.what()};
737 #else
738 // Generic fallback for other compilers
739 throw error{error_code::job_execution_failed, e.what()};
740 #endif
741
742 // We need to provide a return value for non-void tasks
743 if constexpr (std::is_void_v<result_type>) {
744 co_return;
745 } else if constexpr (std::is_same_v<result_type, result_t<void>> ||
746 std::is_same_v<result_type, result_void>) {
747 // For result-based return types
748 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
749 } else if constexpr (detail::is_result_v<result_type>) {
750 // For any other result<T> types
751 co_return result_type(error{error_code::operation_canceled, "Task was cancelled"});
752 } else {
753 // For regular return types - return default value of T
754 co_return static_cast<result_type>(0);
755 }
756 }
757#endif

760// Simple helper function to sleep and resume the coroutine
761inline void sleep_and_resume(std::chrono::milliseconds duration, std::coroutine_handle<> handle) {
762 // Create a shared state to ensure we properly manage thread lifetime
763 struct shared_state {
764 std::thread timer_thread;
765 std::atomic<bool> completed{false};
766 std::atomic<bool> destroyed{false};
767
768 ~shared_state() {
769 destroyed.store(true);
770 if (timer_thread.joinable()) {
771 timer_thread.join();
772 }
773 }
774 };
775
776 // Using shared_ptr for automatic cleanup when all references are gone
777 auto state = std::make_shared<shared_state>();
778
779 // Create the timer thread that will sleep and then resume the coroutine
780 state->timer_thread = std::thread([state, duration, handle]() {
781 std::this_thread::sleep_for(duration);
782
783 // Only resume if we haven't been destroyed yet
784 if (!state->destroyed.load()) {
785 handle.resume();
786 }
787
788 state->completed.store(true);
789 });
790
791#if defined(THREAD_SYSTEM_WINDOWS)
792 // On Windows (including MSVC, MinGW, MSYS), we use a dedicated cleanup approach
793 std::thread cleanup_thread([state]() {
794 // This thread's only job is to hold a reference to the state
795 // until the timer thread completes to ensure proper cleanup
796 while (!state->completed.load()) {
797 std::this_thread::sleep_for(std::chrono::milliseconds(10));
798 }
799 });
800 cleanup_thread.detach();
801#else
802 // On non-Windows platforms, we can safely detach the thread
803 // The shared_ptr will keep the state alive until all references are gone
804 state->timer_thread.detach();
805
806 // Create a second detached thread as a safety net that will
807 // ensure the state lives long enough for the timer thread to finish
808 std::thread([state]() {
809 while (!state->completed.load()) {
810 std::this_thread::sleep_for(std::chrono::milliseconds(50));
811 }
812 }).detach();
813#endif
814}
