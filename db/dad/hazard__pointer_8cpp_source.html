46 hazard_pointer_manager::hazard_pointer_manager(size_t max_threads, size_t pointers_per_thread)
47 : max_threads_(max_threads)
48 , pointers_per_thread_(pointers_per_thread)
49 {
50 // Pre-allocate hazard records
51 for (size_t i = 0; i < max_threads; ++i) {
52 auto* record = new HazardRecord(pointers_per_thread);
53
54 // Add to the list
55 auto* head = head_record_.load(std::memory_order_acquire);
56 do {
57 record->next.store(head, std::memory_order_relaxed);
58 } while (!head_record_.compare_exchange_weak(head, record,
59 std::memory_order_release,
60 std::memory_order_acquire));
61 }
62 }

64 hazard_pointer_manager::~hazard_pointer_manager()
65 {
66 // Force final reclamation
67 scan_and_reclaim();
68
69 // Clean up all hazard records
70 auto* record = head_record_.load(std::memory_order_acquire);
71 while (record) {
72 auto* next = record->next.load(std::memory_order_acquire);
73 delete record;
74 record = next;
75 }
76 }

114 auto hazard_pointer_manager::scan_and_reclaim() -> void
115 {
116 // Collect all hazard pointers
117 auto hazards = collect_hazard_pointers();
118 std::sort(hazards.begin(), hazards.end());
119
120 // Process retired list
121 std::vector<RetiredNode> new_retired_list;
122
123 for (auto& retired : retired_list_) {
124 if (std::binary_search(hazards.begin(), hazards.end(), retired.ptr)) {
125 // Still hazardous, keep in retired list
126 new_retired_list.push_back(std::move(retired));
127 } else {
128 // Safe to reclaim
129 retired.deleter(retired.ptr);
130 total_reclaimed_.fetch_add(1, std::memory_order_relaxed);
131 }
132 }
133
134 retired_list_ = std::move(new_retired_list);
135 last_scan_ = std::chrono::steady_clock::now();
136 }

138 auto hazard_pointer_manager::get_statistics() const -> statistics
139 {
140 statistics stats;
141
142 // Count active hazard pointers
143 size_t active_count = 0;
144 auto* record = head_record_.load(std::memory_order_acquire);
145 while (record) {
146 if (record->owner.load(std::memory_order_acquire) != std::thread::id{}) {
147 for (const auto& hazard : record->hazards) {
148 if (hazard.load(std::memory_order_acquire) != nullptr) {
149 ++active_count;
150 }
151 }
152 }
153 record = record->next.load(std::memory_order_acquire);
154 }
155
156 stats.active_hazard_pointers = active_count;
157 stats.retired_list_size = retired_list_.size();
158 stats.total_reclaimed = total_reclaimed_.load(std::memory_order_relaxed);
159 stats.total_retired = total_retired_.load(std::memory_order_relaxed);
160
161 return stats;
162 }

164 auto hazard_pointer_manager::acquire_record() -> HazardRecord*
165 {
166 // Try to reuse existing record
167 if (local_record_ && local_record_->owner.load(std::memory_order_acquire) == std::this_thread::get_id()) {
168 return local_record_;
169 }
170
171 // Find an unused record
172 auto* record = head_record_.load(std::memory_order_acquire);
173 while (record) {
174 std::thread::id expected{};
175 if (record->owner.compare_exchange_strong(expected, std::this_thread::get_id(),
176 std::memory_order_acquire,
177 std::memory_order_relaxed)) {
178 local_record_ = record;
179 return record;
180 }
181 record = record->next.load(std::memory_order_acquire);
182 }
183
184 // No free records available
185 throw std::runtime_error("No free hazard records available");
186 }

188 auto hazard_pointer_manager::release_record(HazardRecord* record) -> void
189 {
190 if (record) {
191 // Clear all hazard pointers
192 for (auto& hazard : record->hazards) {
193 hazard.store(nullptr, std::memory_order_release);
194 }
195
196 // Release ownership
197 record->owner.store(std::thread::id{}, std::memory_order_release);
198
199 if (local_record_ == record) {
200 local_record_ = nullptr;
201 }
202 }
203 }

205 auto hazard_pointer_manager::acquire_slot() -> std::atomic<void*>*
206 {
207 auto* record = acquire_record();
208
209 // Find an empty slot
210 for (auto& hazard : record->hazards) {
211 void* expected = nullptr;
212 if (hazard.compare_exchange_strong(expected, reinterpret_cast<void*>(1),
213 std::memory_order_acquire,
214 std::memory_order_relaxed)) {
215 hazard.store(nullptr, std::memory_order_release);
216 return &hazard;
217 }
218 }
219
220 // No free slots in record
221 throw std::runtime_error("No free hazard pointer slots");
222 }

242 auto hazard_pointer_manager::collect_hazard_pointers() const -> std::vector<void*>
243 {
244 std::vector<void*> hazards;
245
246 auto* record = head_record_.load(std::memory_order_acquire);
247 while (record) {
248 if (record->owner.load(std::memory_order_acquire) != std::thread::id{}) {
249 for (const auto& hazard : record->hazards) {
250 void* ptr = hazard.load(std::memory_order_acquire);
251 if (ptr && ptr != reinterpret_cast<void*>(1)) {
252 hazards.push_back(ptr);
253 }
254 }
255 }
256 record = record->next.load(std::memory_order_acquire);
257 }
258
259 return hazards;
260 }

262 auto hazard_pointer_manager::should_scan() const -> bool
263 {
264 // Scan if retired list is large
265 if (retired_list_.size() >= RETIRED_THRESHOLD) {
266 return true;
267 }
268
269 // Scan if enough time has passed
270 auto now = std::chrono::steady_clock::now();
271 if (now - last_scan_ >= SCAN_INTERVAL) {
272 return true;
273 }
274
275 return false;
276 }
