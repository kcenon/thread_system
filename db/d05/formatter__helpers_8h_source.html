76 template<typename T>
77 struct string_converter
78 {
79 static auto convert(const T& obj)
80 {
81 if constexpr (requires { obj.to_string(); })
82 {
83 return obj.to_string();
84 }
85 else if constexpr (requires { to_string(obj); })
86 {
87 return std::string(to_string(obj));
88 }
89 else
90 {
91 static_assert(sizeof(T) == 0, "Type does not support string conversion");
92 }
93 }
94 };
