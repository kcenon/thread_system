153 auto log_collector::do_work() -> result_void
154 {
155 // First get a local copy of the log queue with proper synchronization
156 std::shared_ptr<job_queue> queue_copy;
157 {
158 std::lock_guard<std::mutex> lock(queue_mutex_);
159 if (log_queue_ == nullptr)
160 {
161 return error{error_code::resource_allocation_failed, "there is no job_queue"};
162 }
163 queue_copy = log_queue_;
164 }
165
166 // Now work with the local copy
167 auto dequeue_result = queue_copy->dequeue();
168 if (!dequeue_result.has_value())
169 {
170 if (!queue_copy->is_stopped())
171 {
172 return error{error_code::queue_empty,
173 formatter::format("error dequeue job: {}", dequeue_result.get_error().to_string())};
174 }
175
176 return result_void{};
177 }
178
179 auto job_ptr = std::move(dequeue_result.value());
180 if (job_ptr == nullptr)
181 {
182 return error{error_code::job_invalid, "error executing job: received empty job"};
183 }
184
185 auto current_log = std::unique_ptr<log_job>(static_cast<log_job*>(job_ptr.release()));
186
187 auto work_result = current_log->do_work();
188 if (work_result.has_error())
189 {
190 return work_result.get_error();
191 }
192
193 // Get thread-safe copies of the targets and queues
194 log_types console_target_copy;
195 log_types file_target_copy;
196 log_types callback_target_copy;
197 std::weak_ptr<job_queue> console_queue_copy;
198 std::weak_ptr<job_queue> file_queue_copy;
199 std::weak_ptr<job_queue> callback_queue_copy;
200
201 {
202 std::lock_guard<std::mutex> lock(queue_mutex_);
203 console_target_copy = console_log_type_;
204 file_target_copy = file_log_type_;
205 callback_target_copy = callback_log_type_;
206 console_queue_copy = console_queue_;
207 file_queue_copy = file_queue_;
208 callback_queue_copy = callback_queue_;
209 }
210
211 result_void enqueue_result;
212
213 if (current_log->get_type() <= console_target_copy)
214 {
215 enqueue_result = enqueue_log(current_log->get_type(), current_log->get_type(),
216 console_queue_copy, current_log->datetime(),
217 current_log->message());
218 if (enqueue_result.has_error())
219 {
220 return enqueue_result.get_error();
221 }
222 }
223
224 if (current_log->get_type() <= file_target_copy)
225 {
226 enqueue_result = enqueue_log(current_log->get_type(), current_log->get_type(),
227 file_queue_copy, current_log->datetime(),
228 current_log->message());
229 if (enqueue_result.has_error())
230 {
231 return enqueue_result.get_error();
232 }
233 }
234
235 if (current_log->get_type() <= callback_target_copy)
236 {
237 enqueue_result = enqueue_log(current_log->get_type(), current_log->get_type(),
238 callback_queue_copy, current_log->datetime(),
239 current_log->message());
240 if (enqueue_result.has_error())
241 {
242 return enqueue_result.get_error();
243 }
244 }
245
246 return result_void{};
247 }

275 auto log_collector::enqueue_log(const log_types& current_log_type,
276 const log_types& target_log_type,
277 std::weak_ptr<job_queue> weak_queue,
278 const std::string& datetime,
279 const std::string& message) -> result_void
280 {
281 if (current_log_type == log_types::None)
282 {
283 return result_void{};
284 }
285
286 // Safely handle weak pointer - lock and check validity
287 std::shared_ptr<job_queue> locked_queue;
288 {
289 std::lock_guard<std::mutex> lock(queue_mutex_);
290 locked_queue = weak_queue.lock();
291 if (!locked_queue)
292 {
293 return error{error_code::resource_allocation_failed, "Queue is no longer available"};
294 }
295 }
296
297 // Now we have a valid shared_ptr to the queue
298 if (!message.empty())
299 {
300 try
301 {
302 auto job = std::make_unique<message_job>(target_log_type, datetime, message);
303 auto enqueue_result = locked_queue->enqueue(std::move(job));
304 if (enqueue_result.has_error())
305 {
306 return enqueue_result.get_error();
307 }
308 }
309 catch (const std::exception& e)
310 {
311 return error{error_code::job_creation_failed,
312 formatter::format("Error creating or enqueuing message job: {}", e.what())};
313 }
314 }
315
316 return result_void{};
317 }
