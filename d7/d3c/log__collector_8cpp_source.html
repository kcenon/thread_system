124  auto log_collector::do_work() -> std::optional<std::string>
125  {
126  if (log_queue_ == nullptr)
127  {
128  return "there is no job_queue";
129  }
130 
131  auto [job_opt, error] = log_queue_->dequeue();
132  if (error.has_value())
133  {
134  if (!log_queue_->is_stopped())
135  {
136  return formatter::format("error dequeue job: {}", error.value_or("unknown error"));
137  }
138 
139  return std::nullopt;
140  }
141 
142  auto current_job = std::move(job_opt.value());
143  if (current_job == nullptr)
144  {
145  return "error executing job: nullptr";
146  }
147 
148  auto current_log = std::unique_ptr<log_job>(static_cast<log_job*>(current_job.release()));
149 
150  auto work_error = current_log->do_work();
151  if (work_error.has_value())
152  {
153  return std::nullopt;
154  }
155 
156  std::optional<std::string> enqueue_error;
157 
158  if (current_log->get_type() <= console_log_type_)
159  {
160  enqueue_error
161  = enqueue_log(current_log->get_type(), current_log->get_type(), console_queue_,
162  current_log->datetime(), current_log->message());
163  if (enqueue_error.has_value())
164  {
165  return enqueue_error;
166  }
167  }
168 
169  if (current_log->get_type() <= file_log_type_)
170  {
171  enqueue_error
172  = enqueue_log(current_log->get_type(), current_log->get_type(), file_queue_,
173  current_log->datetime(), current_log->message());
174  if (enqueue_error.has_value())
175  {
176  return enqueue_error;
177  }
178  }
179 
180  if (current_log->get_type() <= callback_log_type_)

184  current_log->datetime(), current_log->message());

220  auto log_collector::enqueue_log(const log_types& current_log_type,
221  const log_types& target_log_type,
222  std::weak_ptr<job_queue> weak_queue,
223  const std::string& datetime,
224  const std::string& message) -> std::optional<std::string>
225  {
226  if (current_log_type == log_types::None)
227  {
228  return std::nullopt;
229  }
230 
231  auto locked_queue = weak_queue.lock();
232  if (locked_queue != nullptr && !message.empty())
233  {
234  auto enqueue_error = locked_queue->enqueue(
235  std::make_unique<message_job>(target_log_type, datetime, message));
236  if (enqueue_error.has_value())
237  {
238  return enqueue_error;
239  }
240  }
241 
242  return std::nullopt;
