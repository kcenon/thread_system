102 auto datetime_tool::date(const std::chrono::system_clock::time_point& time,
103 const std::string& separator_character) -> std::string
104 {
105#ifdef USE_STD_CHRONO_CURRENT_ZONE
106 // Modern C++20 approach: Use timezone-aware chrono facilities
107 const auto local_time = std::chrono::current_zone()->to_local(time);
108 const auto ymd
109 = std::chrono::year_month_day{ std::chrono::floor<std::chrono::days>(local_time) };
110
111 const auto year = static_cast<int>(ymd.year());
112 const auto month = static_cast<unsigned>(ymd.month());
113 const auto day = static_cast<unsigned>(ymd.day());
114#else
115 // Legacy fallback: Use traditional C time functions
116 const auto time_t = std::chrono::system_clock::to_time_t(time);
117 const auto* tm = std::localtime(&time_t);
118
119 const auto year = tm->tm_year + 1900; // tm_year is years since 1900
120 const auto month = tm->tm_mon + 1; // tm_mon is 0-based
121 const auto day = tm->tm_mday; // tm_mday is 1-based
122#endif
123
124 return formatter::format("{:04d}{}{:02d}{}{:02d}", year, separator_character, month,
125 separator_character, day);
126 }

152 auto datetime_tool::time(const std::chrono::system_clock::time_point& time,
153 const std::string& separator_character) -> std::string
154 {
155#ifdef USE_STD_CHRONO_CURRENT_ZONE
156 // Modern C++20 approach: Use precise time-of-day extraction
157 const auto local_time = std::chrono::current_zone()->to_local(time);
158 const auto tod = std::chrono::hh_mm_ss{ std::chrono::floor<std::chrono::seconds>(
159 local_time - std::chrono::floor<std::chrono::days>(local_time)) };
160
161 const auto hours = tod.hours().count();
162 const auto minutes = tod.minutes().count();
163 const auto seconds = tod.seconds().count();
164#else
165 // Legacy fallback: Use traditional C time functions
166 const auto time_t = std::chrono::system_clock::to_time_t(time);
167 const auto* tm = std::localtime(&time_t);
168
169 const auto hours = tm->tm_hour; // 24-hour format (0-23)
170 const auto minutes = tm->tm_min; // Minutes (0-59)
171 const auto seconds = tm->tm_sec; // Seconds (0-59)
172#endif
173
174 return formatter::format("{:02d}{}{:02d}{}{:02d}", hours, separator_character, minutes,
175 separator_character, seconds);
176 }

203 auto datetime_tool::milliseconds(const std::chrono::system_clock::time_point& time,
204 const int& milli_portion) -> std::string
205 {
206 const auto duration = time.time_since_epoch();
207 auto milliseconds
208 = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count() % 1000;
209
210 // Apply offset and ensure result stays within valid range
211 milliseconds = (milliseconds + milli_portion) % 1000;
212
213 return formatter::format("{:03d}", milliseconds);
214 }
