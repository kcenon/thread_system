42 auto datetime_tool::date(const std::chrono::system_clock::time_point& time,
43 const std::string& separator_character) -> std::string
44 {
45#ifdef USE_STD_CHRONO_CURRENT_ZONE
46 const auto local_time = std::chrono::current_zone()->to_local(time);
47 const auto ymd
48 = std::chrono::year_month_day{ std::chrono::floor<std::chrono::days>(local_time) };
49
50 const auto year = static_cast<int>(ymd.year());
51 const auto month = static_cast<unsigned>(ymd.month());
52 const auto day = static_cast<unsigned>(ymd.day());
53#else
54 const auto time_t = std::chrono::system_clock::to_time_t(time);
55 const auto* tm = std::localtime(&time_t);
56
57 const auto year = tm->tm_year + 1900;
58 const auto month = tm->tm_mon + 1;
59 const auto day = tm->tm_mday;
60#endif
61
62 return formatter::format("{:04d}{}{:02d}{}{:02d}", year, separator_character, month,
63 separator_character, day);
64 }

66 auto datetime_tool::time(const std::chrono::system_clock::time_point& time,
67 const std::string& separator_character) -> std::string
68 {
69#ifdef USE_STD_CHRONO_CURRENT_ZONE
70 const auto local_time = std::chrono::current_zone()->to_local(time);
71 const auto tod = std::chrono::hh_mm_ss{ std::chrono::floor<std::chrono::seconds>(
72 local_time - std::chrono::floor<std::chrono::days>(local_time)) };
73
74 const auto hours = tod.hours().count();
75 const auto minutes = tod.minutes().count();
76 const auto seconds = tod.seconds().count();
77#else
78 const auto time_t = std::chrono::system_clock::to_time_t(time);
79 const auto* tm = std::localtime(&time_t);
80
81 const auto hours = tm->tm_hour;
82 const auto minutes = tm->tm_min;
83 const auto seconds = tm->tm_sec;
84#endif
85
86 return formatter::format("{:02d}{}{:02d}{}{:02d}", hours, separator_character, minutes,
87 separator_character, seconds);
88 }

90 auto datetime_tool::milliseconds(const std::chrono::system_clock::time_point& time,
91 const int& milli_portion) -> std::string
92 {
93 const auto duration = time.time_since_epoch();
94 auto milliseconds
95 = std::chrono::duration_cast<std::chrono::milliseconds>(duration).count() % 1000;
96
97 milliseconds = (milliseconds + milli_portion) % 1000;
98
99 return formatter::format("{:03d}", milliseconds);
100 }

126 template <typename DurationType, typename ClockType>
127 auto datetime_tool::time_difference(const std::chrono::time_point<ClockType>& start_time,
128 const std::chrono::time_point<ClockType>& end_time)
129 -> double
130 {
131 auto duration
132 = std::chrono::duration<double, typename DurationType::period>(end_time - start_time);
133 return duration.count();
