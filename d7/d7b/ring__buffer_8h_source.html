84 template<typename T>
85 class ring_buffer {
86 public:
95 explicit ring_buffer(std::size_t capacity)
96 : capacity_(capacity + 1) // Reserve one empty slot (for full/empty distinction)
97 , buffer_(capacity_)
98 , head_(0)
99 , tail_(0) {
100 }
101
111 auto push(const T& item) -> bool {
112 const auto current_tail = tail_.load(std::memory_order_relaxed);
113 const auto next_tail = (current_tail + 1) % capacity_;
114
115 if (next_tail == head_.load(std::memory_order_acquire)) {
116 return false; // Buffer is full
117 }
118
119 buffer_[current_tail] = item;
120 tail_.store(next_tail, std::memory_order_release);
121 return true;
122 }
123
133 auto pop(T& item) -> bool {
134 const auto current_head = head_.load(std::memory_order_relaxed);
135
136 if (current_head == tail_.load(std::memory_order_acquire)) {
137 return false; // Buffer is empty
138 }
139
140 item = buffer_[current_head];
141 head_.store((current_head + 1) % capacity_, std::memory_order_release);
142 return true;
143 }
144
152 auto size() const -> std::size_t {
153 const auto current_tail = tail_.load(std::memory_order_acquire);
154 const auto current_head = head_.load(std::memory_order_acquire);
155
156 if (current_tail >= current_head) {
157 return current_tail - current_head;
158 } else {
159 return capacity_ - current_head + current_tail;
160 }
161 }
162
167 auto empty() const -> bool {
168 return head_.load(std::memory_order_acquire) == tail_.load(std::memory_order_acquire);
169 }
170
175 auto full() const -> bool {
176 const auto current_tail = tail_.load(std::memory_order_acquire);
177 const auto next_tail = (current_tail + 1) % capacity_;
178 return next_tail == head_.load(std::memory_order_acquire);
179 }
180
185 auto capacity() const -> std::size_t {
186 return capacity_ - 1; // Actual usable size
187 }
188
199 auto get_recent_items(std::size_t count) const -> std::vector<T> {
200 std::vector<T> result;
201 result.reserve(count);
202
203 const auto current_tail = tail_.load(std::memory_order_acquire);
204 const auto current_head = head_.load(std::memory_order_acquire);
205 const auto current_size = size();
206
207 if (current_size == 0) {
208 return result;
209 }
210
211 const auto items_to_copy = std::min(count, current_size);
212 auto start_index = current_tail;
213
214 // Get items_to_copy items from the back
215 if (items_to_copy <= current_tail) {
216 start_index = current_tail - items_to_copy;
217 } else {
218 start_index = capacity_ - (items_to_copy - current_tail);
219 }
220
221 for (std::size_t i = 0; i < items_to_copy; ++i) {
222 const auto index = (start_index + i) % capacity_;
223 result.push_back(buffer_[index]);
224 }
225
226 return result;
227 }
228
229 private:
230 const std::size_t capacity_;
231 std::vector<T> buffer_;
232 std::atomic<std::size_t> head_;
233 std::atomic<std::size_t> tail_;
234 };

266 template<typename T>
267 class thread_safe_ring_buffer {
268 public:
273 explicit thread_safe_ring_buffer(std::size_t capacity)
274 : capacity_(capacity)
275 , buffer_(capacity)
276 , head_(0)
277 , tail_(0) {
278 }
279
287 auto push(const T& item) -> bool {
288 std::lock_guard<std::mutex> lock(mutex_);
289
290 if ((tail_ + 1) % capacity_ == head_) {
291 return false; // Buffer is full
292 }
293
294 buffer_[tail_] = item;
295 tail_ = (tail_ + 1) % capacity_;
296 return true;
297 }
298
306 auto pop(T& item) -> bool {
307 std::lock_guard<std::mutex> lock(mutex_);
308
309 if (head_ == tail_) {
310 return false; // Buffer is empty
311 }
312
313 item = buffer_[head_];
314 head_ = (head_ + 1) % capacity_;
315 return true;
316 }
317
324 auto size() const -> std::size_t {
325 std::lock_guard<std::mutex> lock(mutex_);
326
327 if (tail_ >= head_) {
328 return tail_ - head_;
329 } else {
330 return capacity_ - head_ + tail_;
331 }
332 }
333
338 auto empty() const -> bool {
339 std::lock_guard<std::mutex> lock(mutex_);
340 return head_ == tail_;
341 }
342
351 auto get_all_items() const -> std::vector<T> {
352 std::lock_guard<std::mutex> lock(mutex_);
353 std::vector<T> result;
354
355 if (head_ == tail_) {
356 return result;
357 }
358
359 result.reserve(size());
360
361 if (tail_ > head_) {
362 for (std::size_t i = head_; i < tail_; ++i) {
363 result.push_back(buffer_[i]);
364 }
365 } else {
366 for (std::size_t i = head_; i < capacity_; ++i) {
367 result.push_back(buffer_[i]);
368 }
369 for (std::size_t i = 0; i < tail_; ++i) {
370 result.push_back(buffer_[i]);
371 }
372 }
373
374 return result;
375 }
376
377 private:
378 const std::size_t capacity_;
379 std::vector<T> buffer_;
380 std::size_t head_;
381 std::size_t tail_;
382 mutable std::mutex mutex_;
383 };
