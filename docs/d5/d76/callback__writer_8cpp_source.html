64  auto callback_writer::do_work() -> std::optional<std::string>
65  {
66  if (job_queue_ == nullptr)
67  {
68  return "there is no job_queue";
69  }
70 
71  auto remaining_logs = job_queue_->dequeue_all();
72  while (!remaining_logs.empty())
73  {
74  auto current_job = std::move(remaining_logs.front());
75  remaining_logs.pop_front();
76 
77  if (callback_ == nullptr)
78  {
79  std::cout << "there is no callback function" << std::endl;
80  continue;
81  }
82 
83  auto current_log
84  = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
85 
86  auto work_error = current_log->do_work();
87  if (work_error.has_value())
88  {
89  std::cout << work_error.value() << std::endl;
90  continue;
91  }
92 
93  callback_(current_log->log_type(), current_log->datetime(), current_log->message());
94  }
95 
96  return std::nullopt;
97  }
