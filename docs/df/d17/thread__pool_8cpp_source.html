99  auto thread_pool::enqueue(std::unique_ptr<thread_worker>&& worker) -> std::optional<std::string>
100  {

104  }
105

108  return "Job queue is null";

111  worker->set_job_queue(job_queue_);

116  if (start_error.has_value())

129  auto thread_pool::stop(const bool& immediately_stop) -> void
130  {
131  if (!start_pool_.load())
132  {
133  return;
134  }
135 
136  if (job_queue_ != nullptr)
137  {
138  job_queue_->stop_waiting_dequeue();
139 
140  if (immediately_stop)
141  {
142  job_queue_->clear();
143  }
144  }
145 
146  for (auto& worker : workers_)
147  {
148  auto stop_error = worker->stop();
149  if (stop_error.has_value())
150  {
151  log_module::write_error("error stopping worker: {}",
152  stop_error.value_or("unknown error"));
153  }
154  }
155 
156  start_pool_.store(false);
157  }
