63  template <typename priority_type>
64  auto priority_thread_pool_t<priority_type>::start(void) -> std::optional<std::string>
65  {
66  if (workers_.empty())
67  {
68  return "no workers to start";
69  }
70 
71  for (auto& worker : workers_)
72  {
73  auto start_error = worker->start();
74  if (start_error.has_value())
75  {
76  stop();
77  return start_error;
78  }
79  }
80 
81  start_pool_.store(true);
82 
83  return std::nullopt;
84  }

116  template <typename priority_type>
117  auto priority_thread_pool_t<priority_type>::enqueue(
118  std::unique_ptr<priority_thread_worker_t<priority_type>>&& worker)
119  -> std::optional<std::string>
120  {
121  if (worker == nullptr)
122  {
123  return "cannot enqueue null worker";
124  }
125 
126  if (job_queue_ == nullptr)
127  {
128  return "cannot enqueue worker due to null job queue";
129  }
130 
131  worker->set_job_queue(job_queue_);
132 
133  if (start_pool_.load())
134  {
135  auto start_error = worker->start();
136  if (start_error.has_value())
137  {
138  stop();
139  return start_error;
140  }
141  }
142 
143  workers_.emplace_back(std::move(worker));
144 
145  return std::nullopt;
146  }

148  template <typename priority_type>
149  auto priority_thread_pool_t<priority_type>::stop(const bool& immediately_stop) -> void
150  {
151  if (!start_pool_.load())
152  {
153  return;
154  }
155 
156  if (job_queue_ != nullptr)
157  {
158  job_queue_->stop_waiting_dequeue();
159 
160  if (immediately_stop)
161  {
162  job_queue_->clear();
163  }
164  }
165 
166  for (auto& worker : workers_)
167  {
168  auto stop_error = worker->stop();
169  if (stop_error.has_value())
170  {
171  log_module::write_error("error stopping worker: {}",
172  stop_error.value_or("unknown error"));
173  }
174  }
175 
176  start_pool_.store(false);
177  }
