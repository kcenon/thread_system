45  auto job_queue::enqueue(std::unique_ptr<job>&& value) -> std::optional<std::string>
46  {
47  if (stop_.load())
48  {
49  return "Job queue is stopped";
50  }
51 
52  if (value == nullptr)
53  {
54  return "cannot enqueue null job";
55  }
56 
57  std::scoped_lock<std::mutex> lock(mutex_);
58 
59  queue_.push_back(std::move(value));
60  queue_size_.fetch_add(1);
61 
62  if (notify_)
63  {
64  condition_.notify_one();
65  }
66 
67  return std::nullopt;
68  }

72  auto job_queue::dequeue()
73  -> std::tuple<std::optional<std::unique_ptr<job>>, std::optional<std::string>>
74  {
75  std::unique_lock<std::mutex> lock(mutex_);
76  condition_.wait(lock, [this]() { return !queue_.empty() || stop_.load(); });
77 
78  if (queue_.empty())
79  {
80  return { std::nullopt, "there are no jobs to dequeue" };
81  }
82 
83  auto value = std::move(queue_.front());
84  queue_.pop_front();
85  queue_size_.fetch_sub(1);
86 
87  return { std::move(value), std::nullopt };
88  }
