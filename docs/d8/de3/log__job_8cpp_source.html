91  auto log_job::do_work() -> std::optional<std::string>
92  {
93  try
94  {
95  datetime_ = formatter::format(
96  "{} {}.{}{}", datetime_tool::date(timestamp_), datetime_tool::time(timestamp_),
97  datetime_tool::milliseconds(timestamp_), datetime_tool::microseconds(timestamp_));
98  std::string converted_message = convert_message();
99 
100  if (!start_time_.has_value())
101  {
102  log_message_ = formatter::format("[{}]", converted_message);
103 
104  return std::nullopt;
105  }
106 
107  auto time_gap = datetime_tool::time_difference<std::chrono::milliseconds,
108  std::chrono::high_resolution_clock>(
109  start_time_.value());
110 
111  log_message_ = formatter::format("[{}] [{} ms]", converted_message, time_gap);
112 
113  return std::nullopt;
114  }
115  catch (const std::exception& e)
116  {
117  return std::string(e.what());
118  }
119  catch (...)
120  {
121  return "unknown error";
122  }
123  }

125  auto log_job::convert_message() const -> std::string
126  {
127  switch (message_type_)
128  {
129  case message_types::String:
130  {
131  return message_;
132  }
133  break;
134  case message_types::WString:
135  {
136  auto [converted, convert_error] = convert_string::to_string(wmessage_);
137  if (!converted.has_value())
138  {
139  return std::string();
140  }
141  return converted.value();
142  }
143  break;
144  default:
145  return std::string();
