60 auto metrics_collector::start() -> thread_module::result_void {
61 if (running_.load(std::memory_order_acquire)) {
62 return thread_module::result_void{thread_module::error{
63 thread_module::error_code::thread_already_running,
64 "Metrics collector is already running"
65 }};
66 }
67
68 stop_requested_.store(false, std::memory_order_release);
69 running_.store(true, std::memory_order_release);
70
71 try {
72 collection_thread_ = std::make_unique<std::thread>(&metrics_collector::collection_loop, this);
73 } catch (const std::exception& e) {
74 running_.store(false, std::memory_order_release);
75 return thread_module::result_void{thread_module::error{
76 thread_module::error_code::thread_start_failure,
77 utility_module::formatter::format("Failed to start collection thread: {}", e.what())
78 }};
79 }
80
81 return {};
82 }
83
84 auto metrics_collector::stop() -> void {
85 if (!running_.load(std::memory_order_acquire)) {
86 return;
87 }
88
89 stop_requested_.store(true, std::memory_order_release);
90
91 if (collection_thread_ && collection_thread_->joinable()) {
92 collection_thread_->join();
93 }
94
95 running_.store(false, std::memory_order_release);
96 collection_thread_.reset();
97 }

146 auto metrics_collector::collection_loop() -> void {
147 while (!stop_requested_.load(std::memory_order_acquire)) {
148 auto start_time = std::chrono::steady_clock::now();
149
150 try {
151 collect_metrics();
152 stats_.total_collections.fetch_add(1, std::memory_order_relaxed);
153 } catch (const std::exception&) {
154 stats_.collection_errors.fetch_add(1, std::memory_order_relaxed);
155 }
156
157 auto end_time = std::chrono::steady_clock::now();
158 auto collection_duration = std::chrono::duration_cast<std::chrono::nanoseconds>(
159 end_time - start_time).count();
160
161 stats_.collection_time_ns.store(static_cast<std::uint64_t>(collection_duration),
162 std::memory_order_relaxed);
163
164 // Wait for collection interval
165 std::this_thread::sleep_for(config_.collection_interval);
166 }
167 }

169 auto metrics_collector::collect_metrics() -> void {
170 if (config_.enable_system_metrics) {
171 collect_system_metrics();
172 }
173 if (config_.enable_thread_pool_metrics) {
174 collect_thread_pool_metrics();
175 }
176 if (config_.enable_worker_metrics) {
177 collect_worker_metrics();
178 }
179
180 // Create and save snapshot
181 auto snapshot = get_current_snapshot();
182 if (!snapshot_buffer_->push(snapshot)) {
183 stats_.buffer_overflows.fetch_add(1, std::memory_order_relaxed);
184 }
185 }

187 auto metrics_collector::collect_system_metrics() -> void {
188 if (!system_metrics_) {
189 return;
190 }
191
192 // Platform-specific system metrics collection
193#ifdef _WIN32
194 // Windows implementation
195 MEMORYSTATUSEX mem_status;
196 mem_status.dwLength = sizeof(mem_status);
197 if (GlobalMemoryStatusEx(&mem_status)) {
198 system_metrics_->memory_usage_bytes.store(
199 mem_status.ullTotalPhys - mem_status.ullAvailPhys,
200 std::memory_order_relaxed
201 );
202 }
203
204 // CPU usage replaced with simple implementation (more complex logic needed in practice)
205 system_metrics_->cpu_usage_percent.store(0, std::memory_order_relaxed);
206
207#elif defined(__linux__)
208 // Linux implementation
209 std::ifstream meminfo("/proc/meminfo");
210 std::string line;
211 std::uint64_t total_mem = 0, free_mem = 0;
212
213 while (std::getline(meminfo, line)) {
214 if (line.find("MemTotal:") == 0) {
215 total_mem = std::stoull(line.substr(line.find_last_of(' ') + 1)) * 1024;
216 } else if (line.find("MemFree:") == 0) {
217 free_mem = std::stoull(line.substr(line.find_last_of(' ') + 1)) * 1024;
218 break;
219 }
220 }
221
222 if (total_mem > 0 && free_mem > 0) {
223 system_metrics_->memory_usage_bytes.store(total_mem - free_mem, std::memory_order_relaxed);
224 }
225
226#elif defined(__APPLE__)
227 // macOS implementation
228 vm_size_t page_size;
229 vm_statistics64_data_t vm_stats;
230 mach_msg_type_number_t info_count = HOST_VM_INFO64_COUNT;
231
232 host_page_size(mach_host_self(), &page_size);
233
234 if (host_statistics64(mach_host_self(), HOST_VM_INFO64,
235 (host_info64_t)&vm_stats, &info_count) == KERN_SUCCESS) {
236
237 std::uint64_t used_memory = (vm_stats.active_count +
238 vm_stats.inactive_count +
239 vm_stats.wire_count) * page_size;
240
241 system_metrics_->memory_usage_bytes.store(used_memory, std::memory_order_relaxed);
242 }
243#endif
244
245 // Active thread count (simple implementation)
246 system_metrics_->active_threads.store(
247 std::thread::hardware_concurrency(),
248 std::memory_order_relaxed
249 );
250
251 system_metrics_->timestamp = std::chrono::steady_clock::now();
252 }
