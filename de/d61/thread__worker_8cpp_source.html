62  auto thread_worker::do_work() -> std::optional<std::string>
63  {
64  if (job_queue_ == nullptr)
65  {
66  return "there is no job_queue";
67  }
68 
69  auto [job_opt, error] = job_queue_->dequeue();
70  if (!job_opt.has_value())
71  {
72  if (!job_queue_->is_stopped())
73  {
74  return formatter::format("error dequeue job: {}", error.value_or("unknown error"));
75  }
76 
77  return std::nullopt;
78  }
79 
80  auto current_job = std::move(job_opt.value());
81  if (current_job == nullptr)
82  {
83  return "error executing job: nullptr";
84  }
85 
86  std::optional<std::chrono::time_point<std::chrono::high_resolution_clock>>
87  started_time_point = std::nullopt;
88  if (use_time_tag_)
89  {
90  started_time_point = std::chrono::high_resolution_clock::now();
91  }
92 
93  current_job->set_job_queue(job_queue_);
94  auto work_error = current_job->do_work();
95  if (work_error.has_value())
96  {
97  return formatter::format("error executing job: {}",
98  work_error.value_or("unknown error"));
99  }
100 
101  if (!started_time_point.has_value())
102  {
103  log_module::write_sequence("job executed successfully: {} on thread_worker",
104  current_job->get_name());
105  }
106 
107  log_module::write_sequence(started_time_point.value(),
108  "job executed successfully: {} on thread_worker",
109  current_job->get_name());
110 
111  return std::nullopt;
112  }
