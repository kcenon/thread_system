93 auto file_writer::do_work() -> result_void
94 {
95 if (job_queue_ == nullptr)
96 {
97 return result_void{error{error_code::resource_allocation_failed, "there is no job_queue"}};
98 }
99
100 if (file_target_ == log_types::None)
101 {
102 return {};
103 }
104
105 check_file_handle();
106
107 auto remaining_logs = job_queue_->dequeue_batch();
108 while (!remaining_logs.empty())
109 {
110 auto current_job = std::move(remaining_logs.front());
111 remaining_logs.pop_front();
112
113 auto current_log
114 = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
115
116 auto work_result = current_log->do_work();
117 if (work_result.has_error())
118 {
119 continue;
120 }
121
122 if (current_log->log_type() == log_types::None)
123 {
124 log_lines_.push_back(formatter::format("[{}]{}", current_log->datetime(),
125 current_log->message(true)));
126
127 continue;
128 }
129
130 log_lines_.push_back(formatter::format("[{}][{}] {}", current_log->datetime(),
131 current_log->log_type(),
132 current_log->message(true)));
133 }
134
135 if (max_lines_ == 0)
136 {
137 log_file_ = write_lines(std::move(log_file_), log_lines_);
138 log_lines_.clear();
139
140 return {};
141 }
142
143 if (log_lines_.size() <= max_lines_)
144 {
145 log_file_ = write_lines(std::move(log_file_), log_lines_);
146 log_file_->close();
147 log_file_.reset();
148
149 return {};
150 }
151
152 size_t index = log_lines_.size() - max_lines_ + 1;
153
154 if (use_backup_)
155 {
156 if (backup_file_ == nullptr)
157 {
158 backup_file_ = std::make_unique<std::fstream>(
159 backup_name_, std::ios_base::out | std::ios_base::app);
160 }
161
162 std::deque<std::string> backup_lines(log_lines_.begin(), log_lines_.begin() + index);
163
164 if (backup_file_ && backup_file_->is_open())
165 {
166 backup_file_->seekp(0, std::ios::end);
167 backup_file_ = write_lines(std::move(backup_file_), backup_lines);
168 }
169 }
170
171 log_lines_.erase(log_lines_.begin(), log_lines_.begin() + index);
172
173 log_file_ = write_lines(std::move(log_file_), log_lines_);
174 log_file_->close();
175 log_file_.reset();
176
177 return {};
178 }

208 auto file_writer::check_file_handle(void) -> void
209 {
210 auto [file_name, backup_name] = generate_file_name();
211
212 if (file_name_ != file_name)
213 {
214 close_file_handle();
215 }
216
217 if (max_lines_ == 0)
218 {
219 if (log_file_ == nullptr)
220 {
221 log_file_ = std::make_unique<std::fstream>(file_name,
222 std::ios_base::out | std::ios_base::app);
223 }
224 }
225 else
226 {
227 if (log_file_ == nullptr)
228 {
229 log_file_ = std::make_unique<std::fstream>(file_name, std::ios_base::out
230 | std::ios_base::trunc);
231 }
232
233 if (use_backup_)
234 {
235 if (backup_file_ == nullptr)
236 {
237 backup_file_ = std::make_unique<std::fstream>(
238 backup_name, std::ios_base::out | std::ios_base::app);
239 }
240 }
241 }
242
243 file_name_ = file_name;
244 backup_name_ = backup_name;
245 }

264 auto file_writer::write_lines(std::unique_ptr<std::fstream> file_handle,
265 const std::deque<std::string>& messages)
266 -> std::unique_ptr<std::fstream>
267 {
268 if (file_handle == nullptr || !file_handle->is_open())
269 {
270 return std::move(file_handle);
271 }
272
273 for (const auto& message : messages)
274 {
275 file_handle->write(message.c_str(), message.size());
276 }
277
278 file_handle->flush();
279
280 return std::move(file_handle);
281 }
