92  auto file_writer::do_work() -> std::optional<std::string>
93  {
94  if (job_queue_ == nullptr)
95  {
96  return "there is no job_queue";
97  }
98 
99  if (file_target_ == log_types::None)
100  {
101  return std::nullopt;
102  }
103 
104  check_file_handle();
105 
106  auto remaining_logs = job_queue_->dequeue_all();
107  while (!remaining_logs.empty())
108  {
109  auto current_job = std::move(remaining_logs.front());
110  remaining_logs.pop_front();
111 
112  auto current_log
113  = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
114 
115  auto work_error = current_log->do_work();
116  if (work_error.has_value())
117  {
118  continue;
119  }
120 
121  if (current_log->log_type() == log_types::None)
122  {
123  log_lines_.push_back(formatter::format("[{}]{}", current_log->datetime(),
124  current_log->message(true)));
125 
126  continue;
127  }
128 
129  log_lines_.push_back(formatter::format("[{}][{}] {}", current_log->datetime(),
130  current_log->log_type(),
131  current_log->message(true)));
132  }
133 
134  if (max_lines_ == 0)
135  {
136  log_file_ = write_lines(std::move(log_file_), log_lines_);
137  log_lines_.clear();
138 
139  return std::nullopt;
140  }
141 
142  if (log_lines_.size() <= max_lines_)
143  {
144  log_file_ = write_lines(std::move(log_file_), log_lines_);
145  log_file_->close();
146  log_file_.reset();
147 
148  return std::nullopt;
149  }
150 
151  size_t index = log_lines_.size() - max_lines_ + 1;
152 
153  if (use_backup_)
154  {
155  if (backup_file_ == nullptr)
156  {
157  backup_file_ = std::make_unique<std::fstream>(
158  backup_name_, std::ios_base::out | std::ios_base::app);
159  }
160 
161  std::deque<std::string> backup_lines(log_lines_.begin(), log_lines_.begin() + index);
162 
163  if (backup_file_ && backup_file_->is_open())
164  {
165  backup_file_->seekp(0, std::ios::end);
166  backup_file_ = write_lines(std::move(backup_file_), backup_lines);
167  }
168  }
169 
170  log_lines_.erase(log_lines_.begin(), log_lines_.begin() + index);
171 
172  log_file_ = write_lines(std::move(log_file_), log_lines_);
173  log_file_->close();
174  log_file_.reset();
175 
176  return std::nullopt;
177  }

264  const std::deque<std::string>& messages)

266  {
