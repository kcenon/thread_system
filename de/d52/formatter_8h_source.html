182 template <typename... FormatArgs>
183 static auto format(const char* formats, const FormatArgs&... args) -> std::string
184 {
185#ifdef USE_STD_FORMAT
186 return std::vformat(formats, std::make_format_args(args...));
187#else
188 // Fallback implementation using ostringstream

190 format_impl(oss, formats, args...);
191 return oss.str();
192#endif
193 }

195 private:
196 // Helper to output arguments with proper conversion
197 template <typename Stream, typename T>
198 static void output_arg(Stream& stream, const T& arg) {
199 if constexpr (std::is_same_v<T, std::string> || std::is_same_v<T, std::string_view> ||
200 std::is_same_v<T, const char*>) {
201 stream << arg;
202 } else if constexpr (std::is_same_v<T, std::wstring> || std::is_same_v<T, std::wstring_view> ||
203 std::is_same_v<T, const wchar_t*>) {
204 stream << arg;
205 } else if constexpr (std::is_array_v<T> && std::is_same_v<std::remove_extent_t<T>, char>) {
206 // Handle char arrays (string literals)
207 stream << arg;
208 } else if constexpr (std::is_array_v<T> && std::is_same_v<std::remove_extent_t<T>, wchar_t>) {
209 // Handle wchar_t arrays (wide string literals)
210 stream << arg;
211 } else if constexpr (std::is_arithmetic_v<T>) {
212 stream << arg;
213 } else if constexpr (std::is_enum_v<T>) {
214 // For enum types, try to use to_string function
215 stream << to_string(arg);
216 } else {
217 // For other custom types with to_string method
218 stream << arg.to_string();
219 }
220 }
221
222 // Simple fallback formatting implementation
223 template <typename Stream>
224 static void format_impl(Stream& stream, const char* format) {
225 stream << format;
226 }
227
228 template <typename Stream, typename T, typename... Args>
229 static void format_impl(Stream& stream, const char* format, const T& arg, const Args&... args) {
230 while (*format) {
231 if (*format == '{' && *(format + 1) == '}') {
232 output_arg(stream, arg);
233 format += 2;
234 format_impl(stream, format, args...);
235 return;
236 }
237 stream << *format++;
238 }
239 }
240
241 // Wide character version
242 template <typename Stream>
243 static void wformat_impl(Stream& stream, const wchar_t* format) {
244 stream << format;
245 }
246
247 template <typename Stream, typename T, typename... Args>
248 static void wformat_impl(Stream& stream, const wchar_t* format, const T& arg, const Args&... args) {
249 while (*format) {

252 format += 2;
253 wformat_impl(stream, format, args...);
254 return;
255 }
256 stream << *format++;
257 }
258 }
