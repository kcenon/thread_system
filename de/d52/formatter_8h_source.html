86 {
87 if constexpr (std::is_same_v<CharT, wchar_t>)
88 {
89 return std::format_to(out, L"{}", Converter{}(value));
90 }
91 else
92 {
93 return std::format_to(out, "{}", Converter{}(value));

120 template <typename FormatContext> auto format(const T& value, FormatContext& context) const
121 {
122 using char_type =
123 typename std::iterator_traits<typename FormatContext::iterator>::value_type;
124 return do_format<char_type>(context.out(), value);
125 }
126 };
