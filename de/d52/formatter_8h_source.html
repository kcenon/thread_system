65 template <typename CharT> static auto do_format(auto& out, const T& value)
66 {
67#ifdef USE_STD_FORMAT
68 if constexpr (std::is_same_v<CharT, wchar_t>)
69 {
70 return std::format_to(out, L"{}", Converter{}(value));
71 }
72 else
73 {
74 return std::format_to(out, "{}", Converter{}(value));
75 }
76#else
77 if constexpr (std::is_same_v<CharT, wchar_t>)
78 {
79 return fmt::format_to(out, L"{}", Converter{}(value));
80 }
81 else
82 {
83 return fmt::format_to(out, "{}", Converter{}(value));
84 }
85#endif
86 }

103 template <typename FormatContext> auto format(const T& value, FormatContext& ctx) const
104 {
105 using char_type =

107 return do_format<char_type>(ctx.out(), value);
