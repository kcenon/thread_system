161 class formatter

174 template <typename... FormatArgs>
175 static auto format(const char* formats, const FormatArgs&... args) -> std::string
176 {
177#ifdef USE_STD_FORMAT
178 return std::vformat(formats, std::make_format_args(args...));
179#else
180 // Fallback implementation using ostringstream
181 std::ostringstream oss;
182 format_impl(oss, formats, args...);
183 return oss.str();
184#endif
185 }
186
187 private:
188 // Simple fallback formatting implementation
189 template <typename Stream>
190 static void format_impl(Stream& stream, const char* format) {
191 stream << format;
192 }
193
194 template <typename Stream, typename T, typename... Args>
195 static void format_impl(Stream& stream, const char* format, const T& arg, const Args&... args) {
196 while (*format) {
197 if (*format == '{' && *(format + 1) == '}') {
198 stream << arg;
199 format += 2;
200 format_impl(stream, format, args...);
201 return;
202 }
203 stream << *format++;
204 }
205 }
206
207 // Wide character version
208 template <typename Stream>
209 static void wformat_impl(Stream& stream, const wchar_t* format) {
210 stream << format;
211 }
212
213 template <typename Stream, typename T, typename... Args>
214 static void wformat_impl(Stream& stream, const wchar_t* format, const T& arg, const Args&... args) {
215 while (*format) {
216 if (*format == L'{' && *(format + 1) == L'}') {
217 stream << arg;
218 format += 2;
219 wformat_impl(stream, format, args...);
220 return;
221 }
222 stream << *format++;
223 }
224 }
