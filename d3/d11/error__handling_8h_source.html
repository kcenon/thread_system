119inline std::string error_code_to_string(error_code code) {
120 using map_entry = std::pair<error_code, std::string_view>;
121 static constexpr map_entry kMap[] = {
122 {error_code::success, "Success"},
123 {error_code::unknown_error, "Unknown error"},
124 {error_code::operation_canceled, "Operation canceled"},
125 {error_code::operation_timeout, "Operation timed out"},
126 {error_code::not_implemented, "Not implemented"},
127 {error_code::invalid_argument, "Invalid argument"},
128 {error_code::thread_already_running, "Thread is already running"},
129 {error_code::thread_not_running, "Thread is not running"},
130 {error_code::thread_start_failure, "Failed to start thread"},
131 {error_code::thread_join_failure, "Failed to join thread"},
132 {error_code::queue_full, "Queue is full"},
133 {error_code::queue_empty, "Queue is empty"},
134 {error_code::queue_stopped, "Queue is stopped"},
135 {error_code::queue_busy, "Queue is busy"},
136 {error_code::job_creation_failed, "Failed to create job"},
137 {error_code::job_execution_failed, "Failed to execute job"},
138 {error_code::job_invalid, "Invalid job"},
139 {error_code::resource_allocation_failed, "Failed to allocate resource"},
140 {error_code::resource_limit_reached, "Resource limit reached"},
141 {error_code::mutex_error, "Mutex error"},
142 {error_code::deadlock_detected, "Deadlock detected"},
143 {error_code::condition_variable_error, "Condition variable error"},
144 {error_code::io_error, "I/O error"},
145 {error_code::file_not_found, "File not found"},
146 };
147 for (const auto& [k, v] : kMap) {
148 if (k == code) return std::string(v);
149 }
150 return std::string("Unknown error code");
151}

634// Helper to convert std::optional<std::string> to result<T>

636result<T> optional_error_to_result(const std::optional<std::string>& error, T&& value) {
637 if (error) {
