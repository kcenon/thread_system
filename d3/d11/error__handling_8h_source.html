100inline std::string error_code_to_string(error_code code) {
101 using map_entry = std::pair<error_code, std::string_view>;
102 static constexpr map_entry kMap[] = {
103 {error_code::success, "Success"},
104 {error_code::unknown_error, "Unknown error"},
105 {error_code::operation_canceled, "Operation canceled"},
106 {error_code::operation_timeout, "Operation timed out"},
107 {error_code::not_implemented, "Not implemented"},
108 {error_code::invalid_argument, "Invalid argument"},
109 {error_code::thread_already_running, "Thread is already running"},
110 {error_code::thread_not_running, "Thread is not running"},
111 {error_code::thread_start_failure, "Failed to start thread"},
112 {error_code::thread_join_failure, "Failed to join thread"},
113 {error_code::queue_full, "Queue is full"},
114 {error_code::queue_empty, "Queue is empty"},
115 {error_code::queue_stopped, "Queue is stopped"},
116 {error_code::job_creation_failed, "Failed to create job"},
117 {error_code::job_execution_failed, "Failed to execute job"},
118 {error_code::job_invalid, "Invalid job"},
119 {error_code::resource_allocation_failed, "Failed to allocate resource"},
120 {error_code::resource_limit_reached, "Resource limit reached"},
121 {error_code::mutex_error, "Mutex error"},
122 {error_code::deadlock_detected, "Deadlock detected"},
123 {error_code::condition_variable_error, "Condition variable error"},
124 {error_code::io_error, "I/O error"},
125 {error_code::file_not_found, "File not found"},
126 };
127 for (const auto& [k, v] : kMap) {
128 if (k == code) return std::string(v);
129 }
130 return std::string("Unknown error code");
131}

166 if (message_.empty()) {
