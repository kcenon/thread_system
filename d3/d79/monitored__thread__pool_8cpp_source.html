191 auto monitored_thread_pool::wrap_job_with_metrics(std::unique_ptr<job>&& original_job)
192 -> std::unique_ptr<job>
193 {
194 // Create a custom job class that wraps the original job with metrics
195 class metrics_wrapper_job : public job
196 {
197 private:
198 std::unique_ptr<job> original_job_;
199 std::shared_ptr<metrics::thread_pool_metrics::job_tracker> tracker_;
200 std::atomic<size_t>* active_workers_;
201
202 public:
203 metrics_wrapper_job(
204 std::unique_ptr<job>&& original_job,
205 std::shared_ptr<metrics::thread_pool_metrics::job_tracker> tracker,
206 std::atomic<size_t>* active_workers)
207 : job("metrics_wrapper")
208 , original_job_(std::move(original_job))
209 , tracker_(tracker)
210 , active_workers_(active_workers)
211 {
212 }
213
214 auto do_work() -> result_void override
215 {
216 // Increment active workers
217 active_workers_->fetch_add(1);
218
219 // Mark job as started
220 tracker_->on_start();
221
222 try
223 {
224 // Execute the original job
225 auto result = original_job_->do_work();
226
227 if (result.has_error())
228 {
229 tracker_->on_error();
230 active_workers_->fetch_sub(1);
231 return result;
232 }
233
234 // Mark job as completed
235 tracker_->on_complete();
236 active_workers_->fetch_sub(1);
237 return result;
238 }
239 catch (...)
240 {
241 tracker_->on_error();
242 active_workers_->fetch_sub(1);
243 throw;
244 }
245 }
246 };
247
248 // Create a metrics tracker for this job
249 auto tracker = std::make_shared<metrics::thread_pool_metrics::job_tracker>(*metrics_);
250
251 return std::make_unique<metrics_wrapper_job>(
252 std::move(original_job), tracker, &active_workers_
253 );
