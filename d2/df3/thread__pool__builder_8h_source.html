201 std::vector<thread_module::cpu_affinity::cpu_set> affinity_sets;
202 if (!config_.per_thread_affinity.empty()) {
203 affinity_sets = config_.per_thread_affinity;
204 } else if (config_.pin_threads_to_cores &&
205 config_.affinity_policy != thread_module::cpu_affinity::affinity_policy::none) {
206 affinity_sets = thread_module::cpu_affinity::generate_affinity_sets(
207 config_.affinity_policy, config_.thread_count, config_.custom_cores);
208 }
209
210 // Create worker threads
211 for (std::size_t i = 0; i < config_.thread_count; ++i) {
212 std::string worker_name = config_.thread_name_prefix + "_" + std::to_string(i);
213 auto worker = std::make_unique<thread_worker>(config_.yield_on_idle);
214
215 // Set the job queue
216 worker->set_job_queue(job_queue);
217
218 // Set CPU affinity if specified
219 if (!affinity_sets.empty() && i < affinity_sets.size()) {
220 worker->set_cpu_affinity(affinity_sets[i]);
221 }
222
223 // Set wake interval if specified
224 if (config_.wake_interval.has_value()) {
225 worker->set_wake_interval(config_.wake_interval);
226 }
227
228 auto result = pool->enqueue(std::move(worker));
229 if (result.has_value()) {
230 return thread_module::error{thread_module::error_code::thread_start_failure, *result};
231 }
232 }
233
234 return pool;
235 } catch (const std::exception& e) {
236 return thread_module::error{thread_module::error_code::resource_allocation_failed, e.what()};
