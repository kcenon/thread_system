174 [[nodiscard]] static auto generate_affinity_sets(
175 affinity_policy policy,
176 size_t thread_count,
177 const std::optional<cpu_set>& custom_cores = std::nullopt)
178 -> std::vector<cpu_set>
179 {
180 std::vector<cpu_set> affinity_sets;
181 affinity_sets.reserve(thread_count);
182
183 const size_t cpu_count = get_cpu_count();
184
185 switch (policy)
186 {
187 case affinity_policy::none:
188 // No affinity - all threads can run on all cores
189 for (size_t i = 0; i < thread_count; ++i)
190 {
191 cpu_set all_cores;
192 for (size_t j = 0; j < cpu_count; ++j)
193 {
194 all_cores.set(j);
195 }
196 affinity_sets.push_back(all_cores);
197 }
198 break;
199
200 case affinity_policy::one_to_one:
201 // Bind each thread to a single core
202 for (size_t i = 0; i < thread_count; ++i)
203 {
204 cpu_set single_core;
205 single_core.set(i % cpu_count);
206 affinity_sets.push_back(single_core);
207 }
208 break;
209
210 case affinity_policy::round_robin:
211 // Distribute threads evenly across all cores
212 {
213 std::vector<size_t> cores_per_thread(cpu_count, 0);
214 for (size_t i = 0; i < thread_count; ++i)
215 {
216 cores_per_thread[i % cpu_count]++;
217 }
218
219 size_t thread_idx = 0;
220 for (size_t core = 0; core < cpu_count; ++core)
221 {
222 for (size_t i = 0; i < cores_per_thread[core]; ++i)
223 {
224 cpu_set cores;
225 cores.set(core);
226 affinity_sets.push_back(cores);
227 thread_idx++;
228 }
229 }
230 }
231 break;
232
233 case affinity_policy::numa_local:
234 // TODO: Implement NUMA-aware affinity
235 // For now, fall back to round_robin
236 return generate_affinity_sets(affinity_policy::round_robin, thread_count);
237
238 case affinity_policy::custom:
239 if (custom_cores.has_value())
240 {
241 for (size_t i = 0; i < thread_count; ++i)
242 {
243 affinity_sets.push_back(custom_cores.value());
244 }
245 }
246 break;
247 }
248
249 return affinity_sets;
250 }

254 static auto set_thread_affinity_windows(std::thread::id thread_id, const cpu_set& cores)
255 -> bool
256 {
257 HANDLE thread_handle = OpenThread(THREAD_SET_INFORMATION | THREAD_QUERY_INFORMATION,
258 FALSE,
259 static_cast<DWORD>(std::hash<std::thread::id>{}(thread_id)));
260 if (!thread_handle)
261 {
262 return false;
263 }
264
265 DWORD_PTR mask = cores.to_ullong();
266 bool result = SetThreadAffinityMask(thread_handle, mask) != 0;
267 CloseHandle(thread_handle);
268 return result;
269 }

271 [[nodiscard]] static auto get_thread_affinity_windows(std::thread::id thread_id)
272 -> std::optional<cpu_set>
273 {
274 HANDLE thread_handle = OpenThread(THREAD_QUERY_INFORMATION,
275 FALSE,
276 static_cast<DWORD>(std::hash<std::thread::id>{}(thread_id)));
277 if (!thread_handle)
278 {
279 return std::nullopt;
280 }
281
282 DWORD_PTR process_mask, system_mask;
283 if (!GetProcessAffinityMask(GetCurrentProcess(), &process_mask, &system_mask))
284 {
285 CloseHandle(thread_handle);
286 return std::nullopt;
287 }
288
289 // Windows doesn't have a direct way to get thread affinity
290 // Return the process affinity mask instead
291 cpu_set cores(process_mask);
292 CloseHandle(thread_handle);
293 return cores;
294 }

296#elif defined(__linux__)
297 static auto set_thread_affinity_linux(std::thread::id thread_id, const cpu_set& cores)
298 -> bool
299 {
300 cpu_set_t cpuset;
301 CPU_ZERO(&cpuset);
302
303 for (size_t i = 0; i < cores.size() && i < CPU_SETSIZE; ++i)
304 {
305 if (cores[i])
306 {
307 CPU_SET(i, &cpuset);
308 }
309 }
310
311 pthread_t thread = pthread_self();
312 if (thread_id == std::this_thread::get_id())
313 {
314 return pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpuset) == 0;
315 }
316
317 // For other threads, we would need to maintain a mapping of thread_id to pthread_t
318 // This is a limitation of the current implementation
319 return false;
320 }

322 [[nodiscard]] static auto get_thread_affinity_linux(std::thread::id thread_id)
323 -> std::optional<cpu_set>
324 {
325 if (thread_id != std::this_thread::get_id())
326 {
327 // Same limitation as above
328 return std::nullopt;
329 }
330
331 cpu_set_t cpuset;
332 CPU_ZERO(&cpuset);
333
334 pthread_t thread = pthread_self();
335 if (pthread_getaffinity_np(thread, sizeof(cpu_set_t), &cpuset) != 0)
336 {
337 return std::nullopt;
338 }
339
340 cpu_set cores;
341 for (size_t i = 0; i < CPU_SETSIZE && i < cores.size(); ++i)
342 {
343 if (CPU_ISSET(i, &cpuset))
344 {
345 cores.set(i);
346 }
347 }
348 return cores;
349 }
350
351#elif defined(__APPLE__)
352 static auto set_thread_affinity_macos(std::thread::id thread_id, const cpu_set& cores)
353 -> bool
354 {
355 // macOS has limited CPU affinity support
356 // We can only set affinity tags, not bind to specific cores
357 if (thread_id != std::this_thread::get_id())
358 {
359 return false;
360 }
361
362 // Find first set core to use as affinity tag
363 size_t affinity_tag = 0;
364 for (size_t i = 0; i < cores.size(); ++i)
365 {
366 if (cores[i])
367 {
368 affinity_tag = i + 1; // Tags start at 1
369 break;
370 }
371 }
372
373 thread_affinity_policy_data_t policy = { static_cast<integer_t>(affinity_tag) };
374 thread_port_t thread_port = pthread_mach_thread_np(pthread_self());
375
376 kern_return_t result = thread_policy_set(thread_port,
377 THREAD_AFFINITY_POLICY,
378 (thread_policy_t)&policy,
379 THREAD_AFFINITY_POLICY_COUNT);
380 return result == KERN_SUCCESS;
381 }
