205  auto logger::start() -> std::optional<std::string>
206  {
207  if (collector_ == nullptr)
208  {
209  return "there is no collector";
210  }
211 
212  collector_->set_console_queue(console_writer_->get_job_queue());
213  collector_->set_file_queue(file_writer_->get_job_queue());
214  collector_->set_callback_queue(callback_writer_->get_job_queue());
215 
216  std::optional<std::string> start_error;
217 
218  start_error = console_writer_->start();
219  if (start_error.has_value())
220  {
221  return formatter::format("cannot start {}: {}", console_writer_->get_thread_title(),
222  start_error.value_or("unknown error"));
223  }
224 
225  start_error = file_writer_->start();
226  if (start_error.has_value())
227  {
228  return formatter::format("cannot start {}: {}", file_writer_->get_thread_title(),
229  start_error.value_or("unknown error"));
230  }
231 
232  start_error = callback_writer_->start();
233  if (start_error.has_value())
234  {
235  return formatter::format("cannot start {}: {}",
236  callback_writer_->get_thread_title(),
237  start_error.value_or("unknown error"));
238  }
239 
240  start_error = collector_->start();
241  if (start_error.has_value())
242  {
243  return formatter::format("cannot start {}: {}", collector_->get_thread_title(),
244  start_error.value_or("unknown error"));
245  }
246 
247  return std::nullopt;
