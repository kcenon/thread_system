212 auto logger::start() -> std::optional<std::string>
213 {
214 std::lock_guard<std::mutex> lock(collector_mutex_);
215 if (collector_ == nullptr)
216 {
217 return "there is no collector";
218 }
219
220 collector_->set_console_queue(console_writer_->get_job_queue());
221 collector_->set_file_queue(file_writer_->get_job_queue());
222 collector_->set_callback_queue(callback_writer_->get_job_queue());
223
224 auto start_result = console_writer_->start();
225 if (start_result.has_error())
226 {
227 return formatter::format("cannot start {}: {}", console_writer_->get_thread_title(),
228 start_result.get_error().to_string());
229 }
230
231 start_result = file_writer_->start();
232 if (start_result.has_error())
233 {
234 return formatter::format("cannot start {}: {}", file_writer_->get_thread_title(),
235 start_result.get_error().to_string());
236 }
237
238 start_result = callback_writer_->start();
239 if (start_result.has_error())
240 {
241 return formatter::format("cannot start {}: {}",
242 callback_writer_->get_thread_title(),
243 start_result.get_error().to_string());
244 }
245
246 start_result = collector_->start();
247 if (start_result.has_error())
248 {
249 return formatter::format("cannot start {}: {}", collector_->get_thread_title(),
250 start_result.get_error().to_string());
251 }
252
253 return std::nullopt;
254 }
