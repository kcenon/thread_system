87 [[nodiscard]] constexpr std::string_view to_string(thread_conditions job_priority)
88 {
89 auto index = static_cast<size_t>(job_priority);
90 return (index < job_detail::thread_conditions_count)
91 ? job_detail::thread_conditions_strings[index]
92 : "UNKNOWN";
93 }
