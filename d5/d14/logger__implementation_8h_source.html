255 template <typename... Args>
256 auto write(const log_types& type, const char* formats, const Args&... args) -> void
257 {
258 std::lock_guard<std::mutex> lock(collector_mutex_);
259 if (collector_ == nullptr)

264 // Check if the message passes the configured target thresholds

266 && collector_->callback_target() < type)
