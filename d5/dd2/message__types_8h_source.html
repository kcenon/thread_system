234 constexpr message_priority to_priority(log_types level) {
235 switch (level) {
236 case log_types::trace:
237 case log_types::debug:
238 return message_priority::low;
239 case log_types::info:
240 case log_types::warn:
241 return message_priority::normal;
242 case log_types::error:
243 return message_priority::high;
244 case log_types::fatal:
245 return message_priority::critical;
246 default:
247 return message_priority::normal;
248 }
249 }

254 struct message_batch {
255 std::vector<log_message> messages;
256 message_priority max_priority = message_priority::low;
257 std::chrono::steady_clock::time_point created_at = std::chrono::steady_clock::now();
258
259 void add_message(log_message msg) {
260 auto priority = to_priority(msg.level);
261 if (priority > max_priority) {
262 max_priority = priority;
263 }
264 messages.emplace_back(std::move(msg));
265 }
266
267 bool empty() const { return messages.empty(); }
268 size_t size() const { return messages.size(); }
269 void clear() {
270 messages.clear();
271 max_priority = message_priority::low;
272 }
273 };
