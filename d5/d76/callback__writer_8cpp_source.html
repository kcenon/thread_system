175 auto callback_writer::do_work() -> result_void
176 {
177 // Validate job queue availability
178 if (job_queue_ == nullptr)
179 {
180 return result_void{error{error_code::resource_allocation_failed, "there is no job_queue"}};
181 }
182
183 // Dequeue all available message jobs for batch processing
184 auto remaining_logs = job_queue_->dequeue_batch();
185 while (!remaining_logs.empty())
186 {
187 auto current_job = std::move(remaining_logs.front());
188 remaining_logs.pop_front();
189
190 // Check if callback function is registered
191 if (callback_ == nullptr)
192 {
193 std::cout << "there is no callback function" << std::endl;
194 continue;
195 }
196
197 // Cast to specific message job type for processing
198 auto current_log
199 = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
200
201 // Validate message job before callback invocation
202 auto work_result = current_log->do_work();
203 if (work_result.has_error())
204 {
205 std::cout << work_result.get_error().to_string() << std::endl;
206 continue;
207 }
208
209 // Invoke registered callback with message data
210 callback_(current_log->log_type(), current_log->datetime(), current_log->message());
211 }
212
213 return {};
214 }
