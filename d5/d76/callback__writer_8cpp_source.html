65 auto callback_writer::do_work() -> result_void
66 {
67 if (job_queue_ == nullptr)
68 {
69 return result_void{error{error_code::resource_allocation_failed, "there is no job_queue"}};
70 }
71
72 auto remaining_logs = job_queue_->dequeue_batch();
73 while (!remaining_logs.empty())
74 {
75 auto current_job = std::move(remaining_logs.front());
76 remaining_logs.pop_front();
77
78 if (callback_ == nullptr)
79 {
80 std::cout << "there is no callback function" << std::endl;
81 continue;
82 }
83
84 auto current_log
85 = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
86
87 auto work_result = current_log->do_work();
88 if (work_result.has_error())
89 {
90 std::cout << work_result.get_error().to_string() << std::endl;
91 continue;
92 }
93
94 callback_(current_log->log_type(), current_log->datetime(), current_log->message());
95 }
96
97 return {};
98 }
