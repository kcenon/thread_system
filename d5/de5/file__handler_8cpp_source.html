87 auto file::remove(const std::string& path) -> std::optional<std::string>
88 {
89 if (!std::filesystem::exists(path))
90 {
91 return "File does not exist";
92 }
93
94 if (!std::filesystem::is_regular_file(path))
95 {
96 return "Path is not a regular file";
97 }
98
99 std::error_code error;
100 if (!std::filesystem::remove(path, error))
101 {
102 return error.message();
103 }
104
105 return std::nullopt;
106 }

131 auto file::load(const std::string& path)
132 -> std::tuple<std::vector<uint8_t>, std::optional<std::string>>
133 {
134 if (!std::filesystem::exists(path))
135 {
136 return { std::vector<uint8_t>{}, "File does not exist" };
137 }
138
139 std::ifstream stream(path, std::ios::binary);
140 if (!stream.is_open())
141 {
142 return { std::vector<uint8_t>{}, "Failed to open file" };
143 }
144
145 // Efficient iterator-based loading
146 std::vector<uint8_t> buffer((std::istreambuf_iterator<char>(stream)),
147 std::istreambuf_iterator<char>());
148 stream.close();
149
150 return { buffer, std::nullopt };
151 }

179 auto file::save(const std::string& path,
180 const std::vector<uint8_t>& data) -> std::optional<std::string>
181 {
182 std::filesystem::path target_path(path);
183 if (!target_path.parent_path().empty())
184 {
185 std::error_code ec;
186 std::filesystem::create_directories(target_path.parent_path(), ec);
187 if (ec)
188 {
189 return "Failed to create directories: " + ec.message();
190 }
191 }
192
193 std::ofstream stream(path, std::ios::binary | std::ios::trunc);
194 if (!stream.is_open())
195 {
196 return "Failed to open file for writing";
197 }
198
199 stream.write(reinterpret_cast<const char*>(data.data()),
200 static_cast<std::streamsize>(data.size()));
201 stream.close();
202
203 return std::nullopt;
204 }
