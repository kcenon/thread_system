47 auto job_queue::enqueue(std::unique_ptr<job>&& value) -> result_void
48 {
49 if (stop_.load())
50 {
51 return error{error_code::queue_stopped, "Job queue is stopped"};
52 }
53
54 if (value == nullptr)
55 {
56 return error{error_code::invalid_argument, "cannot enqueue null job"};
57 }
58
59 std::scoped_lock<std::mutex> lock(mutex_);
60
61 queue_.push_back(std::move(value));
62
63 if (notify_)
64 {
65 condition_.notify_one();
66 }
67
68 return {};
69 }

71 auto job_queue::enqueue_batch(std::vector<std::unique_ptr<job>>&& jobs) -> result_void
72 {
73 if (stop_.load())
74 {
75 return error{error_code::queue_stopped, "Job queue is stopped"};
76 }
77
78 if (jobs.empty())
79 {
80 return error{error_code::invalid_argument, "cannot enqueue empty batch"};
81 }
82
83 for (auto& job : jobs)
84 {
85 if (job == nullptr)
86 {
87 return error{error_code::invalid_argument, "cannot enqueue null job in batch"};
88 }
89 }
90
91 std::scoped_lock<std::mutex> lock(mutex_);
92
93 for (auto& job : jobs)
94 {
95 queue_.push_back(std::move(job));
96 }
97
98 if (notify_)
99 {
100 condition_.notify_one();
101 }
102
103 return {};
104 }

106 auto job_queue::dequeue() -> result<std::unique_ptr<job>>
107 {
108 std::unique_lock<std::mutex> lock(mutex_);
109 condition_.wait(lock, [this]() { return !queue_.empty() || stop_.load(); });
110
111 if (queue_.empty())
112 {
113 return error{error_code::queue_empty, "there are no jobs to dequeue"};
114 }
115
116 auto value = std::move(queue_.front());
117 queue_.pop_front();
118
119 return value;
120 }
