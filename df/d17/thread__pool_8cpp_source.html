54 auto thread_pool::start(void) -> std::optional<std::string>
55 {
56 if (workers_.empty())
57 {
58 return "No workers to start";
59 }
60
61 for (auto& worker : workers_)
62 {
63 auto start_result = worker->start();
64 if (start_result.has_error())
65 {
66 stop();
67 return start_result.get_error().to_string();
68 }
69 }
70
71 start_pool_.store(true);
72
73 return std::nullopt;
74 }

121 auto thread_pool::enqueue(std::unique_ptr<thread_worker>&& worker) -> std::optional<std::string>
122 {

125 return "Worker is null";
126 }
127
128 if (job_queue_ == nullptr)
129 {
130 return "Job queue is null";

132

183 auto thread_pool::stop(const bool& immediately_stop) -> void
184 {
185 if (!start_pool_.load())
186 {
187 return;
188 }
189
190 if (job_queue_ != nullptr)
191 {
192 job_queue_->stop_waiting_dequeue();
193
194 if (immediately_stop)
195 {
196 job_queue_->clear();
197 }
198 }
199
200 for (auto& worker : workers_)
201 {
202 auto stop_result = worker->stop();
203 if (stop_result.has_error())
204 {
205 log_module::write_error("error stopping worker: {}",
206 stop_result.get_error().to_string());
207 }
208 }
209
210 start_pool_.store(false);
211 }

213 auto thread_pool::to_string(void) const -> std::string
214 {
215 std::string format_string;
216
217 formatter::format_to(std::back_inserter(format_string), "{} is {},\n", thread_title_,
218 start_pool_.load() ? "running" : "stopped");
219 formatter::format_to(std::back_inserter(format_string), "\tjob_queue: {}\n\n",
220 (job_queue_ != nullptr ? job_queue_->to_string() : "nullptr"));
221 formatter::format_to(std::back_inserter(format_string), "\tworkers: {}\n", workers_.size());
222 for (const auto& worker : workers_)
223 {
224 formatter::format_to(std::back_inserter(format_string), "\t{}\n", worker->to_string());
225 }
226
227 return format_string;
