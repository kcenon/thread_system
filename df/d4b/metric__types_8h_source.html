115 const std::string name_;

119 gauge(std::string name, std::string desc)
120 : name_(std::move(name))
121 , description_(std::move(desc)) {}
122
127 void set(T value) noexcept {
128 value_.store(value, std::memory_order_relaxed);
129 update_min_max(value);
130 }
131
136 void increment(T delta = 1) noexcept {
137 T new_value = value_.fetch_add(delta, std::memory_order_relaxed) + delta;
138 update_min_max(new_value);
139 }
140
145 void decrement(T delta = 1) noexcept {
146 T new_value = value_.fetch_sub(delta, std::memory_order_relaxed) - delta;
147 update_min_max(new_value);
148 }
149
154 T get() const noexcept {
155 return value_.load(std::memory_order_relaxed);
156 }
157
158 json to_json() const override {
159 return {
160 {"type", "gauge"},
161 {"name", name_},
162 {"description", description_},
163 {"value", get()},
164 {"min", min_value_.load(std::memory_order_relaxed)},
165 {"max", max_value_.load(std::memory_order_relaxed)}
166 };
167 }
168
169 std::string type_name() const override { return "gauge"; }
170 const std::string& name() const override { return name_; }
171 const std::string& description() const override { return description_; }
172
173private:
174 void update_min_max(T value) noexcept {
175 // Lock-free min update
176 T current_min = min_value_.load(std::memory_order_relaxed);
177 while (value < current_min &&
178 !min_value_.compare_exchange_weak(current_min, value,
179 std::memory_order_relaxed));
180
181 // Lock-free max update
182 T current_max = max_value_.load(std::memory_order_relaxed);
183 while (value > current_max &&
184 !max_value_.compare_exchange_weak(current_max, value,
185 std::memory_order_relaxed));
186 }

194template<typename T = double, size_t BucketCount = 64>
195class histogram : public metric_interface {
196private:
197 struct bucket {
198 alignas(64) std::atomic<uint64_t> count{0};
199 alignas(64) std::atomic<T> sum{0};
200 };
201
202 std::array<bucket, BucketCount> buckets_;
203 std::vector<T> bucket_boundaries_;
204 alignas(64) std::atomic<uint64_t> total_count_{0};
205 alignas(64) std::atomic<T> total_sum_{0};
206 const std::string name_;
207 const std::string description_;
208 mutable std::shared_mutex boundaries_mutex_;
209
210public:
211 histogram(std::string name, std::string desc,
212 std::vector<T> boundaries = {})
213 : name_(std::move(name))
214 , description_(std::move(desc))
215 , bucket_boundaries_(std::move(boundaries)) {
216
217 if (bucket_boundaries_.empty()) {
218 // Default exponential buckets: 0.001, 0.002, 0.004, ... seconds
219 generate_exponential_buckets();
220 }
221
222 // Ensure boundaries are sorted
223 std::sort(bucket_boundaries_.begin(), bucket_boundaries_.end());
224 }
225
230 void observe(T value) noexcept {
231 total_count_.fetch_add(1, std::memory_order_relaxed);
232 total_sum_.fetch_add(value, std::memory_order_relaxed);
233
234 // Find appropriate bucket
235 size_t bucket_idx = find_bucket(value);
236 if (bucket_idx < buckets_.size()) {
237 buckets_[bucket_idx].count.fetch_add(1, std::memory_order_relaxed);
238 buckets_[bucket_idx].sum.fetch_add(value, std::memory_order_relaxed);
239 }
240 }
241
247 T percentile(double p) const {
248 std::shared_lock lock(boundaries_mutex_);
249
250 uint64_t total = total_count_.load(std::memory_order_relaxed);
251 if (total == 0) return 0;
252
253 uint64_t target = static_cast<uint64_t>(total * p);
254 uint64_t current = 0;
255
256 for (size_t i = 0; i < bucket_boundaries_.size(); ++i) {
257 current += buckets_[i].count.load(std::memory_order_relaxed);
258 if (current >= target) {
259 return bucket_boundaries_[i];
260 }
261 }
262
263 return bucket_boundaries_.empty() ? 0 : bucket_boundaries_.back();
264 }
265
266 json to_json() const override {
267 std::shared_lock lock(boundaries_mutex_);
268
269 uint64_t total_count = total_count_.load(std::memory_order_relaxed);
270 T total_sum = total_sum_.load(std::memory_order_relaxed);
271
272 json buckets_json = json::array();
273 for (size_t i = 0; i < bucket_boundaries_.size(); ++i) {
274 uint64_t count = buckets_[i].count.load(std::memory_order_relaxed);
275 if (count > 0) {
276 buckets_json.push_back({
277 {"le", bucket_boundaries_[i]},
278 {"count", count}
279 });
280 }
281 }
282
283 return {
284 {"type", "histogram"},
285 {"name", name_},
286 {"description", description_},
287 {"count", total_count},
288 {"sum", total_sum},
289 {"mean", total_count > 0 ? total_sum / total_count : 0},
290 {"p50", percentile(0.50)},
291 {"p90", percentile(0.90)},
292 {"p95", percentile(0.95)},
293 {"p99", percentile(0.99)},
294 {"p999", percentile(0.999)},
295 {"buckets", buckets_json}
296 };
297 }
298
299 std::string type_name() const override { return "histogram"; }
300 const std::string& name() const override { return name_; }
301 const std::string& description() const override { return description_; }
302
303private:
304 void generate_exponential_buckets() {
305 bucket_boundaries_.clear();
306 T value = 0.001; // Start at 1ms
307 for (size_t i = 0; i < std::min(size_t(20), BucketCount - 1); ++i) {
308 bucket_boundaries_.push_back(value);
309 value *= 2;
310 }
311 bucket_boundaries_.push_back(std::numeric_limits<T>::max());
312 }
313
314 size_t find_bucket(T value) const noexcept {
315 // Binary search for efficiency
316 auto it = std::lower_bound(bucket_boundaries_.begin(),
317 bucket_boundaries_.end(), value);
318 return std::distance(bucket_boundaries_.begin(), it);
319 }

326template<typename T = double>
327class summary : public metric_interface {
328private:
329 struct time_window {
330 std::chrono::steady_clock::time_point timestamp;
331 T value;
332 };
333
334 mutable std::shared_mutex mutex_;
335 std::deque<time_window> values_;
336 const std::chrono::seconds window_duration_;
337 const std::string name_;
338 const std::string description_;
339
340public:
341 summary(std::string name, std::string desc,
342 std::chrono::seconds window = std::chrono::seconds(300))
343 : window_duration_(window)
344 , name_(std::move(name))
345 , description_(std::move(desc)) {}
346
351 void observe(T value) {
352 auto now = std::chrono::steady_clock::now();
353
354 std::unique_lock lock(mutex_);
355 values_.push_back({now, value});
356
357 // Remove old values outside the window
358 auto cutoff = now - window_duration_;
359 while (!values_.empty() && values_.front().timestamp < cutoff) {
360 values_.pop_front();
361 }
362 }
363
364 json to_json() const override {
365 std::shared_lock lock(mutex_);
366
367 if (values_.empty()) {
368 return {
369 {"type", "summary"},
370 {"name", name_},
371 {"description", description_},
372 {"window_seconds", window_duration_.count()},
373 {"count", 0}
374 };
375 }
376
377 // Calculate statistics
378 std::vector<T> sorted_values;
379 sorted_values.reserve(values_.size());
380
381 T sum = 0;
382 for (const auto& tw : values_) {
383 sorted_values.push_back(tw.value);
384 sum += tw.value;
385 }
386
387 T mean = sum / values_.size();
388
389 // Sort for percentiles
390 std::sort(sorted_values.begin(), sorted_values.end());
391
392 auto percentile = [&sorted_values](double p) -> T {
393 size_t idx = static_cast<size_t>(sorted_values.size() * p);
394 return sorted_values[std::min(idx, sorted_values.size() - 1)];
395 };
396
397 return {
398 {"type", "summary"},
399 {"name", name_},
400 {"description", description_},
401 {"window_seconds", window_duration_.count()},
402 {"count", values_.size()},
403 {"mean", mean},
404 {"p50", percentile(0.50)},
405 {"p90", percentile(0.90)},
406 {"p95", percentile(0.95)},
407 {"p99", percentile(0.99)},
408 {"p999", percentile(0.999)}
409 };
410 }
411
412 std::string type_name() const override { return "summary"; }
413 const std::string& name() const override { return name_; }
414 const std::string& description() const override { return description_; }
415};
