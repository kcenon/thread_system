64 constexpr const char* get_level_color(log_types level) {
65 switch (level) {
66 case log_types::trace: return colors::dim;
67 case log_types::debug: return colors::cyan;
68 case log_types::info: return colors::green;
69 case log_types::warn: return colors::yellow;
70 case log_types::error: return colors::red;
71 case log_types::fatal: return colors::bright_red;
72 default: return colors::reset;
73 }
