95 constexpr const char* get_level_color(log_types level) {
96 switch (level) {
97 case log_types::trace: return colors::dim;
98 case log_types::debug: return colors::cyan;
99 case log_types::info: return colors::green;
100 case log_types::warn: return colors::yellow;
101 case log_types::error: return colors::red;
102 case log_types::fatal: return colors::bright_red;
103 default: return colors::reset;
104 }
105 }

110 inline std::string format_colored_level(log_types level, bool use_color = true) {
111 std::string level_str = to_string(level);
112
113 if (!use_color) {
114 return level_str;
115 }
116
117 return std::string(get_level_color(level)) + level_str + colors::reset;
118 }
