58  auto thread_base::start(void) -> std::optional<std::string>
59  {
60 #ifdef USE_STD_JTHREAD
61  if (stop_source_.has_value())
62 #else
63  if (worker_thread_ && worker_thread_->joinable())
64 #endif
65  {
66  return "thread is already running";
67  }
68 
69  stop();
70 
71 #ifdef USE_STD_JTHREAD
72  stop_source_ = std::stop_source();
73 #else
74  stop_requested_ = false;
75 #endif
76 
77  try
78  {
79 #ifdef USE_STD_JTHREAD
80  worker_thread_ = std::make_unique<std::jthread>(
81 #else
82  worker_thread_ = std::make_unique<std::thread>(
83 #endif
84  [this](void)
85  {
86 #ifdef USE_STD_JTHREAD
87  auto stop_token = stop_source_.value().get_token();
88 #endif
89 
90  std::optional<std::string> work_error;
91  work_error = before_start();
92  if (work_error.has_value())
93  {
94  std::cerr << "error before start: " << work_error.value_or("unknown error")
95  << std::endl;
96  }
97 
98 #ifdef USE_STD_JTHREAD
99  while (!stop_token.stop_requested() || should_continue_work())
100 #else
101  while (!stop_requested_ || should_continue_work())
102 #endif
103  {
104  thread_condition_.store(thread_conditions::Waiting);
105 
106  std::unique_lock<std::mutex> lock(cv_mutex_);
107  if (wake_interval_.has_value())
108  {
109 #ifdef USE_STD_JTHREAD
110  worker_condition_.wait_for(
111  lock, wake_interval_.value(), [this, &stop_token]()
112  { return stop_token.stop_requested() || should_continue_work(); });
113 #else
114  worker_condition_.wait_for(
115  lock, wake_interval_.value(),
116  [this]() { return stop_requested_ || should_continue_work(); });
117 #endif
118  }
119  else
120  {
121 #ifdef USE_STD_JTHREAD
122  worker_condition_.wait(
123  lock, [this, &stop_token]()
124  { return stop_token.stop_requested() || should_continue_work(); });
125 #else
126  worker_condition_.wait(
127  lock,
128  [this]() { return stop_requested_ || should_continue_work(); });
129 #endif
130  }
131 
132 #ifdef USE_STD_JTHREAD
133  if (stop_token.stop_requested() && !should_continue_work())
134 #else
135  if (stop_requested_ && !should_continue_work())
136 #endif
137  {
138  thread_condition_.store(thread_conditions::Stopping);
139 
140  break;
141  }
142 
143  try
144  {
145  thread_condition_.store(thread_conditions::Working);
146 
147  work_error = do_work();
148  if (work_error.has_value())
149  {
150  std::cerr << "error doing work on " << thread_title_ << " : "
151  << work_error.value_or("unknown error") << std::endl;
152  }
153  }
154  catch (const std::exception& e)
155  {
156  std::cerr << e.what() << '\n';
157  }
158  }
159 
160  work_error = after_stop();
161  if (work_error.has_value())
162  {
163  std::cerr << "error after stop: " << work_error.value_or("unknown error")
164  << std::endl;
165  }
166  });
167  }
168  catch (const std::bad_alloc& e)
169  {
170 #ifdef USE_STD_JTHREAD
171  stop_source_.reset();
172 #else
173  stop_requested_ = true;
174 #endif
175 
176  worker_thread_.reset();
177 
178  return e.what();
179  }
180 
181  return std::nullopt;
182  }

184  auto thread_base::stop(void) -> std::optional<std::string>
185  {
186  if (worker_thread_ == nullptr)
187  {
188  return "thread is not running";
189  }
190 
191  if (worker_thread_->joinable())
192  {
193 #ifdef USE_STD_JTHREAD
194  if (stop_source_.has_value())
195  {
196  stop_source_.value().request_stop();
197  }
198 #else
199  stop_requested_ = true;
200 #endif
201 
202  {
203  std::scoped_lock<std::mutex> lock(cv_mutex_);
204  worker_condition_.notify_all();
205  }
206 
207  worker_thread_->join();
208  }
209 
210 #ifdef USE_STD_JTHREAD
211  stop_source_.reset();
212 #endif
213  worker_thread_.reset();
214 
215  thread_condition_.store(thread_conditions::Stopped);
216 
217  return std::nullopt;
218  }
