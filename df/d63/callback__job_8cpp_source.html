66 auto callback_job::do_work(void) -> result_void
67 {
68 // If we have a data callback, use it with the stored data
69 if (data_callback_)
70 {
71 return data_callback_(data_);
72 }
73 // Otherwise use the standard callback
74 else if (callback_)
75 {
76 return callback_();
77 }
78 // If no callbacks are set, delegate to the base class
79 else
80 {
81 return job::do_work();
82 }
83 }
