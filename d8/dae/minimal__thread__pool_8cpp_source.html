38#include "thread_pool.h"

42using namespace thread_pool_module;
43using namespace thread_module;
44
45int main() {
46 std::cout << "=== Minimal Thread Pool Sample (No Logger) ===" << std::endl;
47
48 // Create thread pool
49 auto pool = std::make_shared<thread_pool>("MinimalPool");
50
51 // Create workers
52 const size_t worker_count = 4;
53 std::vector<std::unique_ptr<thread_worker>> workers;
54
55 for (size_t i = 0; i < worker_count; ++i) {
56 workers.push_back(std::make_unique<thread_worker>(false));
57 }
58
59 // Add workers to pool
60 auto result = pool->enqueue_batch(std::move(workers));
61 if (result.has_error()) {
62 std::cerr << "Error adding workers: " << result.get_error().to_string() << std::endl;
63 return 1;
64 }
65
66 // Start the pool
67 result = pool->start();
68 if (result.has_error()) {
69 std::cerr << "Error starting pool: " << result.get_error().to_string() << std::endl;
70 return 1;
71 }
72
73 std::cout << "Thread pool started with " << worker_count << " workers" << std::endl;
74
75 // Submit some jobs
76 std::atomic<int> completed_jobs{0};
77 const int total_jobs = 20;
78
79 std::cout << "Submitting " << total_jobs << " jobs..." << std::endl;
80
81 for (int i = 0; i < total_jobs; ++i) {
82 auto job = std::make_unique<callback_job>(
83 [i, &completed_jobs]() -> result_void {
84 // Simulate some work
85 std::this_thread::sleep_for(std::chrono::milliseconds(100));
86
87 // Print progress (thread-safe)
88 int current = completed_jobs.fetch_add(1) + 1;
89 std::cout << "Job " << i << " completed. Total: "
90 << current << "/" << total_jobs << std::endl;
91
92 return result_void{};
93 }
94 );
95
96 result = pool->enqueue(std::move(job));
97 if (result.has_error()) {
98 std::cerr << "Error enqueuing job: " << result.get_error().to_string() << std::endl;
99 }
100 }
101
102 // Wait for all jobs to complete
103 std::cout << "Waiting for jobs to complete..." << std::endl;
104 while (completed_jobs.load() < total_jobs) {
105 std::this_thread::sleep_for(std::chrono::milliseconds(100));
106 }
107
108 std::cout << "All jobs completed!" << std::endl;
109
110 // Stop the pool
111 auto stop_result = pool->stop();
112 if (stop_result.has_error()) {
113 std::cerr << "Error stopping pool: " << stop_result.get_error().to_string() << std::endl;
114 }
115 std::cout << "Thread pool stopped." << std::endl;
116
117 return 0;
118}
