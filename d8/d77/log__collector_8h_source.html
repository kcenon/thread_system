202 template <typename StringType>
203 auto write_string(const log_types& type,
204 const StringType& message,
205 std::optional<std::chrono::time_point<std::chrono::high_resolution_clock>>
206 start_time) -> void
207 {
208 std::unique_ptr<log_job> new_log_job;
209
210 try
211 {
212 new_log_job = std::make_unique<log_job>(message, type, start_time);
213 }
214 catch (const std::bad_alloc& e)
215 {
216 std::cerr << "error allocating log job: " << e.what() << std::endl;
217 return;
218 }
219
220 // Add thread safety - lock before accessing shared resource
221 std::lock_guard<std::mutex> lock(queue_mutex_);
222 if (log_queue_ == nullptr)
223 {
224 std::cerr << "Cannot enqueue log: queue is null\n";
225 return;
226 }
227
228 auto enqueue_result = log_queue_->enqueue(std::move(new_log_job));
229 if (enqueue_result.has_error())
230 {
231 std::cerr << formatter::format("error enqueuing log job: {}\n",
232 enqueue_result.get_error().to_string());
233 }
234 }
