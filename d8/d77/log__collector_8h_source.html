198 template <typename StringType>
199 auto write_string(const log_types& type,
200 const StringType& message,
201 std::optional<std::chrono::time_point<std::chrono::high_resolution_clock>>
202 start_time) -> void
203 {
204 std::unique_ptr<log_job> new_log_job;
205
206 try
207 {
208 new_log_job = std::make_unique<log_job>(message, type, start_time);
209 }
210 catch (const std::bad_alloc& e)
211 {
212 std::cerr << "error allocating log job: " << e.what() << std::endl;
213 return;
214 }
215
216 // Add thread safety - lock before accessing shared resource
217 std::lock_guard<std::mutex> lock(queue_mutex_);
218 if (log_queue_ == nullptr)
219 {
220 std::cerr << "Cannot enqueue log: queue is null\n";
221 return;
222 }
223
224 auto enqueue_result = log_queue_->enqueue(std::move(new_log_job));
225 if (enqueue_result.has_error())
226 {
227 std::cerr << formatter::format("error enqueuing log job: {}\n",
228 enqueue_result.get_error().to_string());
229 }
230 }
