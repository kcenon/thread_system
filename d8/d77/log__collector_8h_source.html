210 template <typename StringType>

213 std::optional<std::chrono::time_point<std::chrono::high_resolution_clock>>
214 start_time) -> void
215 {
216 std::unique_ptr<log_job> new_log_job;
217
218 try
219 {
220 new_log_job = std::make_unique<log_job>(message, type, start_time);
221 }
222 catch (const std::bad_alloc& e)
223 {
224 std::cerr << "error allocating log job: " << e.what() << std::endl;
225 return;
226 }
227
228 auto enqueue_error = log_queue_->enqueue(std::move(new_log_job));
229 if (enqueue_error.has_value())
230 {
231 std::cerr << formatter::format("error enqueuing log job: {}\n",
232 enqueue_error.value_or("unknown error"));
233 }
234 }
