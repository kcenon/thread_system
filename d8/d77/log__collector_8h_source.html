225 {
226 std::unique_ptr<log_job> new_log_job;
227
228 try
229 {
230 new_log_job = std::make_unique<log_job>(message, type, start_time);
231 }
232 catch (const std::bad_alloc& e)
233 {
234 std::cerr << "error allocating log job: " << e.what() << std::endl;
235 return;
236 }
237
238 // Add thread safety - lock before accessing shared resource
239 std::lock_guard<std::mutex> lock(queue_mutex_);
240 if (log_queue_ == nullptr)
241 {
242 std::cerr << "Cannot enqueue log: queue is null\n";
243 return;
244 }
245
246 auto enqueue_result = log_queue_->enqueue(std::move(new_log_job));
247 if (enqueue_result.has_error())
248 {
249 std::cerr << formatter::format("error enqueuing log job: {}\n",
250 enqueue_result.get_error().to_string());
251 }
252 }
