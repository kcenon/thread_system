70 auto lockfree_log_collector::should_continue_work() const -> bool
71 {
72 // Use atomic flag for quick check first
73 if (has_messages_.load(std::memory_order_acquire))
74 {
75 return true;
76 }
77
78 // Double-check with actual queue
79 return lockfree_log_queue_ && !lockfree_log_queue_->empty();
80 }

95 auto lockfree_log_collector::do_work() -> result_void
96 {
97 if (!lockfree_log_queue_)
98 {
99 return result_void(error(error_code::invalid_argument,
100 "Lock-free log queue is null"));
101 }
102
103 // Process multiple messages in batch for better performance
104 constexpr size_t batch_size = 32;
105 size_t processed = 0;
106
107 while (processed < batch_size)
108 {
109 auto dequeue_result = lockfree_log_queue_->dequeue();
110 if (!dequeue_result.has_value())
111 {
112 // No more messages
113 break;
114 }
115
116 auto job = std::move(dequeue_result.value());
117 if (auto log_job_ptr = dynamic_cast<class log_job*>(job.get()))
118 {
119 // Format timestamp
120 auto now = std::chrono::system_clock::now();
121 auto time_t_now = std::chrono::system_clock::to_time_t(now);
122
123 std::tm tm_now;
124 #ifdef _WIN32
125 localtime_s(&tm_now, &time_t_now);
126 #else
127 localtime_r(&time_t_now, &tm_now);
128 #endif
129
130 std::ostringstream datetime_stream;
131 datetime_stream << std::put_time(&tm_now, "%Y-%m-%d %H:%M:%S");
132 auto datetime = datetime_stream.str();
133
134 // Get message details
135 auto log_type = log_job_ptr->get_type();
136 auto message = log_job_ptr->message();
137
138 // Use parent class method to enqueue to writer queues
139 // We need to access these through getter methods since they're private
140 log_collector::enqueue_log(log_type, console_target(),
141 get_console_queue(), datetime, message);
142
143 log_collector::enqueue_log(log_type, file_target(),
144 get_file_queue(), datetime, message);
145
146 log_collector::enqueue_log(log_type, callback_target(),
147 get_callback_queue(), datetime, message);
148 }
149
150 ++processed;
151 }
152
153 // Update atomic flag
154 if (lockfree_log_queue_->empty())
155 {
156 has_messages_.store(false, std::memory_order_release);
157 }
158
159 return {};
160 }
