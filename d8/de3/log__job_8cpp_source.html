92 auto log_job::do_work() -> result_void
93 {
94 try
95 {
96 datetime_ = formatter::format(
97 "{} {}.{}{}", datetime_tool::date(timestamp_), datetime_tool::time(timestamp_),
98 datetime_tool::milliseconds(timestamp_), datetime_tool::microseconds(timestamp_));
99 std::string converted_message = convert_message();
100
101 if (!start_time_.has_value())
102 {
103 log_message_ = formatter::format("[{}]", converted_message);
104 return result_void{};
105 }
106
107 auto time_gap = datetime_tool::time_difference<std::chrono::milliseconds,
108 std::chrono::high_resolution_clock>(
109 start_time_.value());
110
111 log_message_ = formatter::format("[{}] [{} ms]", converted_message, time_gap);
112
113 return result_void{};
114 }
115 catch (const std::exception& e)
116 {
117 return error{error_code::job_execution_failed, e.what()};
118 }
119 catch (...)
120 {
121 return error{error_code::job_execution_failed, "unknown error"};
122 }
123 }

125 auto log_job::convert_message() const -> std::string
126 {
127 switch (message_type_)
128 {
129 case message_types::String:
130 {
131 return message_;
132 }
133 break;
134 case message_types::WString:
135 {
136 auto [converted, convert_error] = convert_string::to_string(wmessage_);
137 if (!converted.has_value())
138 {
139 return std::string();
140 }
141 return converted.value();
142 }
143 break;
144 default:
145 return std::string();
146 }
147 }
