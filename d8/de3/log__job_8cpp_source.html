192 auto log_job::do_work() -> result_void
193 {
194 try
195 {
196 // Format timestamp with full precision for accurate logging
197 datetime_ = formatter::format(
198 "{} {}.{}{}", datetime_tool::date(timestamp_), datetime_tool::time(timestamp_),
199 datetime_tool::milliseconds(timestamp_), datetime_tool::microseconds(timestamp_));
200
201 // Convert message to appropriate string encoding
202 std::string converted_message = convert_message();
203
204 // Format without timing information if no start time provided
205 if (!start_time_.has_value())
206 {
207 log_message_ = formatter::format("[{}]", converted_message);
208 return result_void{};
209 }
210
211 // Calculate performance timing in milliseconds
212 auto time_gap = datetime_tool::time_difference<std::chrono::milliseconds,
213 std::chrono::high_resolution_clock>(
214 start_time_.value());
215
216 // Format with timing information included
217 log_message_ = formatter::format("[{}] [{} ms]", converted_message, time_gap);
218
219 return result_void{};
220 }
221 catch (const std::exception& e)
222 {
223 return error{error_code::job_execution_failed, e.what()};
224 }
225 catch (...)
226 {
227 return error{error_code::job_execution_failed, "unknown error"};
228 }
229 }

231 auto log_job::convert_message() const -> std::string
232 {
233 switch (message_type_)
234 {
235 case message_types::String:
236 {
237 return message_;
238 }
239 break;
240 case message_types::WString:
241 {
242 auto [converted, convert_error] = convert_string::to_string(wmessage_);
243 if (!converted.has_value())
244 {
245 return std::string();
246 }
247 return converted.value();
248 }
249 break;
250 default:
251 return std::string();
252 }
253 }
