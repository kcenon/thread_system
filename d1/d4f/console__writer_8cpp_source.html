67  auto console_writer::do_work() -> std::optional<std::string>
68  {
69  if (job_queue_ == nullptr)
70  {
71  return "there is no job_queue";
72  }
73 
74  std::string console_buffer = "";
75  auto remaining_logs = job_queue_->dequeue_all();
76  while (!remaining_logs.empty())
77  {
78  auto current_job = std::move(remaining_logs.front());
79  remaining_logs.pop_front();
80 
81  auto current_log
82  = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
83 
84  auto work_error = current_log->do_work();
85  if (work_error.has_value())
86  {
87  continue;
88  }
89 
90  if (current_log->log_type() == log_types::None)
91  {
92  formatter::format_to(std::back_inserter(console_buffer), "[{}]{}",
93  current_log->datetime(), current_log->message(true));
94 
95  continue;
96  }
97 
98  formatter::format_to(std::back_inserter(console_buffer), "[{}][{}] {}",
99  current_log->datetime(), current_log->log_type(),
100  current_log->message(true));
101  }
102 
103 #ifdef USE_STD_FORMAT
104  std::cout << console_buffer;
105 #else
106  fmt::print("{}", console_buffer);
107 #endif
108 
109  return std::nullopt;
110  }
