131 auto console_writer::do_work() -> result_void
132 {
133 if (job_queue_ == nullptr)
134 {
135 return result_void{error{error_code::resource_allocation_failed, "there is no job_queue"}};
136 }
137
138 std::string console_buffer = "";
139 auto remaining_logs = job_queue_->dequeue_batch();
140 while (!remaining_logs.empty())
141 {
142 auto current_job = std::move(remaining_logs.front());
143 remaining_logs.pop_front();
144
145 auto current_log
146 = std::unique_ptr<message_job>(static_cast<message_job*>(current_job.release()));
147
148 auto work_result = current_log->do_work();
149 if (work_result.has_error())
150 {
151 continue;
152 }
153
154 if (current_log->log_type() == log_types::None)
155 {
156 formatter::format_to(std::back_inserter(console_buffer), "[{}]{}",
157 current_log->datetime(), current_log->message(true));
158
159 continue;
160 }
161
162 formatter::format_to(std::back_inserter(console_buffer), "[{}][{}] {}",
163 current_log->datetime(), current_log->log_type(),
164 current_log->message(true));
165 }
166
167#ifdef USE_STD_FORMAT
168 std::cout << console_buffer;
169#else
170 fmt::print("{}", console_buffer);
171#endif
172
173 return {};
174 }
