84 [[nodiscard]] constexpr std::string_view to_string(log_types log_type)
85 {
86 auto index = static_cast<size_t>(log_type);
87 return (index < log_detail::log_type_count) ? log_detail::log_type_strings[index]
88 : "UNKNOWN";
